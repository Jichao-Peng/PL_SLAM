134     void BinaryDescriptor::setWidthOfBand( int width )
135     {
136       params.widthOfBand_ = width;
137     
138       /* reserve enough space for EDLine objects and images in Gaussian pyramid */
139       edLineVec_.resize( params.numOfOctave_ );
140       images_sizes.resize( params.numOfOctave_ );
141     
142       for ( int i = 0; i < params.numOfOctave_; i++ )
143         edLineVec_[i] = Ptr < EDLineDetector > ( new EDLineDetector() );
144     
145       /* prepare a vector to host local weights F_l*/
146       gaussCoefL_.resize( params.widthOfBand_ * 3 );
147     
148       /* compute center of central band (every computation involves 2-3 bands) */
149       double u = ( params.widthOfBand_ * 3 - 1 ) / 2;
150     
151       /* compute exponential part of F_l */
152       double sigma = ( params.widthOfBand_ * 2 + 1 ) / 2;  // (widthOfBand_*2+1)/2;
153       double invsigma2 = -1 / ( 2 * sigma * sigma );
154     
155       /* compute all local weights */
156       double dis;
157       for ( int i = 0; i < params.widthOfBand_ * 3; i++ )
158       {
159         dis = i - u;
160         gaussCoefL_[i] = exp( dis * dis * invsigma2 );
161       }
162     
163       /* prepare a vector for global weights F_g*/
164       gaussCoefG_.resize( NUM_OF_BANDS * params.widthOfBand_ );
165     
166       /* compute center of LSR */
167       u = ( NUM_OF_BANDS * params.widthOfBand_ - 1 ) / 2;
168     
169       /* compute exponential part of F_g */
170       sigma = u;
171       invsigma2 = -1 / ( 2 * sigma * sigma );
172       for ( int i = 0; i < NUM_OF_BANDS * params.widthOfBand_; i++ )
173       {
174         dis = i - u;
175         gaussCoefG_[i] = exp( dis * dis * invsigma2 );
176       }
177     }

349     /* compute Gaussian pyramids */
350     void BinaryDescriptor::computeGaussianPyramid( const Mat& image, const int numOctaves )
351     {
352       /* clear class fields */
353       images_sizes.clear();
354       octaveImages.clear();
355     
356       /* insert input image into pyramid */
357       cv::Mat currentMat = image.clone();
358       cv::GaussianBlur( currentMat, currentMat, cv::Size( 5, 5 ), 1 );
359       octaveImages.push_back( currentMat );
360       images_sizes.push_back( currentMat.size() );
361     
362       /* fill Gaussian pyramid */
363       for ( int pyrCounter = 1; pyrCounter < numOctaves; pyrCounter++ )
364       {
365         /* compute and store next image in pyramid and its size */
366         pyrDown( currentMat, currentMat, Size( currentMat.cols / params.reductionRatio, currentMat.rows / params.reductionRatio ) );
367         octaveImages.push_back( currentMat );
368         images_sizes.push_back( currentMat.size() );
369       }
370     }

373     void BinaryDescriptor::computeSobel( const cv::Mat& image, const int numOctaves )
374     {
375     
376       /* compute Gaussian pyramids */
377       computeGaussianPyramid( image, numOctaves );
378     
379       /* reinitialize class structures */
380       dxImg_vector.clear();
381       dyImg_vector.clear();
382     
383     //  dxImg_vector.resize( params.numOfOctave_ );
384     //  dyImg_vector.resize( params.numOfOctave_ );
385     
386       dxImg_vector.resize( octaveImages.size() );
387       dyImg_vector.resize( octaveImages.size() );

392         dxImg_vector[sobelCnt].create( images_sizes[sobelCnt].height, images_sizes[sobelCnt].width, CV_16SC1 );
393         dyImg_vector[sobelCnt].create( images_sizes[sobelCnt].height, images_sizes[sobelCnt].width, CV_16SC1 );

395         cv::Sobel( octaveImages[sobelCnt], dxImg_vector[sobelCnt], CV_16SC1, 1, 0, 3 );
396         cv::Sobel( octaveImages[sobelCnt], dyImg_vector[sobelCnt], CV_16SC1, 0, 1, 3 );

416     {
417

452     void BinaryDescriptor::detectImpl( const Mat& imageSrc, std::vector<KeyLine>& keylines, const Mat& mask ) const
453     {
454     
455       cv::Mat image;
456       if( imageSrc.channels() != 1 )
457       {
458         cvtColor( imageSrc, image, COLOR_BGR2GRAY );
459       }
460       else
461         image = imageSrc.clone();
462     
463       /*check whether image depth is different from 0 */
464       if( image.depth() != 0 )
465         throw std::runtime_error( "Warning, depth image!= 0" );
466     
467       /* create a pointer to self */
468       BinaryDescriptor *bn = const_cast<BinaryDescriptor*>( this );
469     
470       /* detect and arrange lines across octaves */
471       ScaleLines sl;
472       bn->OctaveKeyLines( image, sl );
473     
474       /* fill KeyLines vector */
475       for ( int i = 0; i < (int) sl.size(); i++ )
476       {
477         for ( size_t j = 0; j < sl[i].size(); j++ )
478         {
479           /* get current line */
480           OctaveSingleLine osl = sl[i][j];
481     
482           /* create a KeyLine object */
483           KeyLine kl;
484     
485           /* fill KeyLine's fields */
486           kl.startPointX = osl.startPointX;  //extremes[0];
487           kl.startPointY = osl.startPointY;  //extremes[1];
488           kl.endPointX = osl.endPointX;  //extremes[2];
489           kl.endPointY = osl.endPointY;  //extremes[3];
490           kl.sPointInOctaveX = osl.sPointInOctaveX;
491           kl.sPointInOctaveY = osl.sPointInOctaveY;
492           kl.ePointInOctaveX = osl.ePointInOctaveX;
493           kl.ePointInOctaveY = osl.ePointInOctaveY;
494           kl.lineLength = osl.lineLength;
495           kl.numOfPixels = osl.numOfPixels;
496     
497           kl.angle = osl.direction;
498           kl.class_id = i;
499           kl.octave = osl.octaveCount;
500           kl.size = ( osl.endPointX - osl.startPointX ) * ( osl.endPointY - osl.startPointY );
501           kl.response = osl.lineLength / max( images_sizes[osl.octaveCount].width, images_sizes[osl.octaveCount].height );
502           kl.pt = Point2f( ( osl.endPointX + osl.startPointX ) / 2, ( osl.endPointY + osl.startPointY ) / 2 );
503     
504           /* store KeyLine */
505           keylines.push_back( kl );
506         }
507     
508       }
509     
510       /* delete undesired KeyLines, according to input mask */
511       if( !mask.empty() )
512       {
513         for ( size_t keyCounter = 0; keyCounter < keylines.size(); keyCounter++ )
514         {
515           KeyLine kl = keylines[keyCounter];
516           if( mask.at < uchar > ( (int) kl.startPointY, (int) kl.startPointX ) == 0 && mask.at < uchar > ( (int) kl.endPointY, (int) kl.endPointX ) == 0 )
517             keylines.erase( keylines.begin() + keyCounter );
518         }
519       }
520     
521     }

538     /* implementation of descriptors computation */
539     void BinaryDescriptor::computeImpl( const Mat& imageSrc, std::vector<KeyLine>& keylines, Mat& descriptors, bool returnFloatDescr,
540                                         bool useDetectionData ) const
541     {
542     
543       /* convert input image to gray scale */
544       cv::Mat image;
545     
546       if( imageSrc.channels() != 1 )
547         cvtColor( imageSrc, image, COLOR_BGR2GRAY );
548       else
549         image = imageSrc.clone();
550     
551       /*check whether image's depth is different from 0 */
552       if( image.depth() != 0 )
553         throw std::runtime_error( "Error, depth of image != 0" );
554     
555       /* keypoints list can't be empty */
556       if( keylines.size() == 0 )
557       {
558         std::cout << "Error: keypoint list is empty" << std::endl;
559         return;
560       }
561     
562       BinaryDescriptor* bd = const_cast<BinaryDescriptor*>( this );
563     
564       /* get maximum class_id and octave*/
565       int numLines = 0;
566       int octaveIndex = -1;
567       for ( size_t l = 0; l < keylines.size(); l++ )
568       {
569         if( keylines[l].class_id > numLines )
570           numLines = keylines[l].class_id;
571     
572         if( keylines[l].octave > octaveIndex )
573           octaveIndex = keylines[l].octave;
574       }
575     
576       if( !useDetectionData )
577         bd->computeSobel( image, octaveIndex + 1 );
578     
579       /* create a ScaleLines object */
580       OctaveSingleLine fictiousOSL;
581     //  fictiousOSL.octaveCount = params.numOfOctave_ + 1;
582     //  LinesVec lv( params.numOfOctave_, fictiousOSL );
583       fictiousOSL.octaveCount = octaveIndex + 1;
584       LinesVec lv( octaveIndex + 1, fictiousOSL );
585       ScaleLines sl( numLines + 1, lv );
586     
587       /* create a map to record association between KeyLines and their position
588        in ScaleLines vector */
589       std::map<std::pair<int, int>, size_t> correspondences;
590     
591       /* fill ScaleLines object */
592       for ( size_t slCounter = 0; slCounter < keylines.size(); slCounter++ )
593       {
594         /* get a KeyLine object and create a new line */
595         KeyLine kl = keylines[slCounter];
596         OctaveSingleLine osl;
597     
598         /* insert data in newly created line */
599         osl.startPointX = kl.startPointX;
600         osl.startPointY = kl.startPointY;
601         osl.endPointX = kl.endPointX;
602         osl.endPointY = kl.endPointY;
603         osl.sPointInOctaveX = kl.sPointInOctaveX;
604         osl.sPointInOctaveY = kl.sPointInOctaveY;
605         osl.ePointInOctaveX = kl.ePointInOctaveX;
606         osl.ePointInOctaveY = kl.ePointInOctaveY;
607         osl.lineLength = kl.lineLength;
608         osl.numOfPixels = kl.numOfPixels;
609         osl.salience = kl.response;
610     
611         osl.direction = kl.angle;
612         osl.octaveCount = kl.octave;
613     
614         /* store new line */
615         sl[kl.class_id][kl.octave] = osl;
616     
617         /* update map */
618         int id = kl.class_id;
619         int oct = kl.octave;
620         correspondences.insert( std::pair<std::pair<int, int>, size_t>( std::pair<int, int>( id, oct ), slCounter ) );
621       }
622     
623       /* delete useless OctaveSingleLines */
624       for ( size_t i = 0; i < sl.size(); i++ )
625       {
626         for ( size_t j = 0; j < sl[i].size(); j++ )
627         {
628           //if( (int) ( sl[i][j] ).octaveCount > params.numOfOctave_ )
629           if( (int) ( sl[i][j] ).octaveCount > octaveIndex )
630             ( sl[i] ).erase( ( sl[i] ).begin() + j );
631         }
632       }
633     
634       /* compute LBD descriptors */
635       bd->computeLBD( sl, useDetectionData );
636     
637       /* resize output matrix */
638       if( !returnFloatDescr )
639         descriptors = cv::Mat( (int) keylines.size(), 32, CV_8UC1 );
640     
641       else
642         descriptors = cv::Mat( (int) keylines.size(), NUM_OF_BANDS * 8, CV_32FC1 );
643     
644       /* fill output matrix with descriptors */
645       for ( int k = 0; k < (int) sl.size(); k++ )
646       {
647         for ( int lineC = 0; lineC < (int) sl[k].size(); lineC++ )
648         {
649           /* get original index of keypoint */
650           int lineOctave = ( sl[k][lineC] ).octaveCount;
651           int originalIndex = (int)correspondences.find( std::pair<int, int>( k, lineOctave ) )->second;
652     
653           if( !returnFloatDescr )
654           {
655             /* get a pointer to correspondent row in output matrix */
656             uchar* pointerToRow = descriptors.ptr( originalIndex );
657     
658             /* get LBD data */
659             float* desVec = &sl[k][lineC].descriptor.front();
660     
661             /* fill current row with binary descriptor */
662             for ( int comb = 0; comb < 32; comb++ )
663             {
664               *pointerToRow = bd->binaryConversion( &desVec[8 * combinations[comb][0]], &desVec[8 * combinations[comb][1]] );
665               pointerToRow++;
666             }
667           }
668     
669           else
670           {
671             /* get a pointer to correspondent row in output matrix */
672             float* pointerToRow = descriptors.ptr<float>( originalIndex );
673     
674             /* get LBD data */
675             std::vector<float> desVec = sl[k][lineC].descriptor;
676     
677             for ( int count = 0; count < (int) desVec.size(); count++ )
678             {
679               *pointerToRow = desVec[count];
680               pointerToRow++;
681             }
682           }
683     
684         }
685       }

689     int BinaryDescriptor::OctaveKeyLines( cv::Mat& image, ScaleLines &keyLines )
690     {
691     
692       /* final number of extracted lines */
693       unsigned int numOfFinalLine = 0;
694     
695       /* sigma values and reduction factor used in Gaussian pyramids */
696       float preSigma2 = 0;  //orignal image is not blurred, has zero sigma;
697       float curSigma2 = 1.0;  //[sqrt(2)]^0=1;
698       double factor = sqrt( 2.0 );  //the down sample factor between connective two octave images
699     
700       /* loop over number of octaves */
701       for ( int octaveCount = 0; octaveCount < params.numOfOctave_; octaveCount++ )
702       {
703         /* matrix storing results from blurring processes */
704         cv::Mat blur;
705     
706         /* apply Gaussian blur */
707         float increaseSigma = sqrt( curSigma2 - preSigma2 );
708         cv::GaussianBlur( image, blur, cv::Size( params.ksize_, params.ksize_ ), increaseSigma );
709         images_sizes[octaveCount] = blur.size();
710     
711         /* for current octave, extract lines */
712         if( ( edLineVec_[octaveCount]->EDline( blur ) ) != 1 )
713         {
714           return -1;
715         }
716     
717         /* update number of total extracted lines */
718         numOfFinalLine += edLineVec_[octaveCount]->lines_.numOfLines;
719     
720         /* resize image for next level of pyramid */
721         cv::resize( blur, image, cv::Size(), ( 1.f / factor ), ( 1.f / factor ) );
722     
723         /* update sigma values */
724         preSigma2 = curSigma2;
725         curSigma2 = curSigma2 * 2;
726     
727       } /* end of loop over number of octaves */
728     
729       /* prepare a vector to store octave information associated to extracted lines */
730       std::vector < OctaveLine > octaveLines( numOfFinalLine );
731     
732       /* set lines' counter to 0 for reuse */
733       numOfFinalLine = 0;
734     
735       /* counter to give a unique ID to lines in LineVecs */
736       unsigned int lineIDInScaleLineVec = 0;
737     
738       /* floats to compute lines' lengths */
739       float dx, dy;
740     
741       /* loop over lines extracted from scale 0 (original image) */
742       for ( unsigned int lineCurId = 0; lineCurId < edLineVec_[0]->lines_.numOfLines; lineCurId++ )
743       {
744         /* FOR CURRENT LINE: */
745     
746         /* set octave from which it was extracted */
747         octaveLines[numOfFinalLine].octaveCount = 0;
748         /* set ID within its octave */
749         octaveLines[numOfFinalLine].lineIDInOctave = lineCurId;
750         /* set a unique ID among all lines extracted in all octaves */
751         octaveLines[numOfFinalLine].lineIDInScaleLineVec = lineIDInScaleLineVec;
752     
753         /* compute absolute value of difference between X coordinates of line's extreme points */
754         dx = fabs( edLineVec_[0]->lineEndpoints_[lineCurId][0] - edLineVec_[0]->lineEndpoints_[lineCurId][2] );
755         /* compute absolute value of difference between Y coordinates of line's extreme points */
756         dy = fabs( edLineVec_[0]->lineEndpoints_[lineCurId][1] - edLineVec_[0]->lineEndpoints_[lineCurId][3] );
757         /* compute line's length */
758         octaveLines[numOfFinalLine].lineLength = sqrt( dx * dx + dy * dy );
759     
760         /* update counters */
761         numOfFinalLine++;
762         lineIDInScaleLineVec++;
763       }
764     
765       /* create and fill an array to store scale factors */
766       float *scale = new float[params.numOfOctave_];
767       scale[0] = 1;
768       for ( int octaveCount = 1; octaveCount < params.numOfOctave_; octaveCount++ )
769       {
770         scale[octaveCount] = (float) ( factor * scale[octaveCount - 1] );
771       }
772     
773       /* some variables' declarations */
774       float rho1, rho2, tempValue;
775       float direction, diffNear, length;
776       unsigned int octaveID, lineIDInOctave;
777     
778       /*more than one octave image, organize lines in scale space.
779        *lines corresponding to the same line in octave images should have the same index in the ScaleLineVec */
780       if( params.numOfOctave_ > 1 )
781       {
782         /* some other variables' declarations */
783         double twoPI = 2 * M_PI;
784         unsigned int closeLineID = 0;
785         float endPointDis, minEndPointDis, minLocalDis, maxLocalDis;
786         float lp0, lp1, lp2, lp3, np0, np1, np2, np3;
787     
788         /* loop over list of octaves */
789         for ( int octaveCount = 1; octaveCount < params.numOfOctave_; octaveCount++ )
790         {
791           /*for each line in current octave image, find their corresponding lines in the octaveLines,
792            *give them the same value of lineIDInScaleLineVec*/
793     
794           /* loop over list of lines extracted from current octave */
795           for ( unsigned int lineCurId = 0; lineCurId < edLineVec_[octaveCount]->lines_.numOfLines; lineCurId++ )
796           {
797             /* get (scaled) known term from equation of current line */
798             rho1 = (float) ( scale[octaveCount] * fabs( edLineVec_[octaveCount]->lineEquations_[lineCurId][2] ) );
799     
800             /*nearThreshold depends on the distance of the image coordinate origin to current line.
801              *so nearThreshold = rho1 * nearThresholdRatio, where nearThresholdRatio = 1-cos(10*pi/180) = 0.0152*/
802             tempValue = (float) ( rho1 * 0.0152 );
803             float diffNearThreshold = ( tempValue > 6 ) ? ( tempValue ) : 6;
804             diffNearThreshold = ( diffNearThreshold < 12 ) ? diffNearThreshold : 12;
805     
806             /* compute scaled lenght of current line */
807             dx = fabs( edLineVec_[octaveCount]->lineEndpoints_[lineCurId][0] - edLineVec_[octaveCount]->lineEndpoints_[lineCurId][2] );  //x1-x2
808             dy = fabs( edLineVec_[octaveCount]->lineEndpoints_[lineCurId][1] - edLineVec_[octaveCount]->lineEndpoints_[lineCurId][3] );  //y1-y2
809             length = scale[octaveCount] * sqrt( dx * dx + dy * dy );
810     
811             minEndPointDis = 12;
812             /* loop over the octave representations of all lines */
813             for ( unsigned int lineNextId = 0; lineNextId < numOfFinalLine; lineNextId++ )
814             {
815               /* if a line from same octave is encountered,
816                a comparison with it shouldn't be considered */
817               octaveID = octaveLines[lineNextId].octaveCount;
818               if( (int) octaveID == octaveCount )
819               {  //lines in the same layer of octave image should not be compared.
820                 break;
821               }
822     
823               /* take ID in octave of line to be compared */
824               lineIDInOctave = octaveLines[lineNextId].lineIDInOctave;
825     
826               /*first check whether current line and next line are parallel.
827                *If line1:a1*x+b1*y+c1=0 and line2:a2*x+b2*y+c2=0 are parallel, then
828                *-a1/b1=-a2/b2, i.e., a1b2=b1a2.
829                *we define parallel=fabs(a1b2-b1a2)
830                *note that, in EDLine class, we have normalized the line equations
831                *to make a1^2+ b1^2 = a2^2+ b2^2 = 1*/
832               direction = fabs( edLineVec_[octaveCount]->lineDirection_[lineCurId] - edLineVec_[octaveID]->lineDirection_[lineIDInOctave] );
833     
834               /* the angle between two lines are larger than 10degrees
835                (i.e. 10*pi/180=0.1745), they are not close to parallel */
836               if( direction > 0.1745 && ( twoPI - direction > 0.1745 ) )
837               {
838                 continue;
839               }
840               /*now check whether current line and next line are near to each other.
841                *If line1:a1*x+b1*y+c1=0 and line2:a2*x+b2*y+c2=0 are near in image, then
842                *rho1 = |a1*0+b1*0+c1|/sqrt(a1^2+b1^2) and rho2 = |a2*0+b2*0+c2|/sqrt(a2^2+b2^2) should close.
843                *In our case, rho1 = |c1| and rho2 = |c2|, because sqrt(a1^2+b1^2) = sqrt(a2^2+b2^2) = 1;
844                *note that, lines are in different octave images, so we define near =  fabs(scale*rho1 - rho2) or
845                *where scale is the scale factor between to octave images*/
846     
847               /* get known term from equation to be compared */
848               rho2 = (float) ( scale[octaveID] * fabs( edLineVec_[octaveID]->lineEquations_[lineIDInOctave][2] ) );
849               /* compute difference between known ters */
850               diffNear = fabs( rho1 - rho2 );
851     
852               /* two lines are not close in the image */
853               if( diffNear > diffNearThreshold )
854               {
855                 continue;
856               }
857     
858               /*now check the end points distance between two lines, the scale of  distance is in the original image size.
859                * find the minimal and maximal end points distance*/
860     
861               /* get the extreme points of the two lines */
862               lp0 = scale[octaveCount] * edLineVec_[octaveCount]->lineEndpoints_[lineCurId][0];
863               lp1 = scale[octaveCount] * edLineVec_[octaveCount]->lineEndpoints_[lineCurId][1];
864               lp2 = scale[octaveCount] * edLineVec_[octaveCount]->lineEndpoints_[lineCurId][2];
865               lp3 = scale[octaveCount] * edLineVec_[octaveCount]->lineEndpoints_[lineCurId][3];
866               np0 = scale[octaveID] * edLineVec_[octaveID]->lineEndpoints_[lineIDInOctave][0];
867               np1 = scale[octaveID] * edLineVec_[octaveID]->lineEndpoints_[lineIDInOctave][1];
868               np2 = scale[octaveID] * edLineVec_[octaveID]->lineEndpoints_[lineIDInOctave][2];
869               np3 = scale[octaveID] * edLineVec_[octaveID]->lineEndpoints_[lineIDInOctave][3];
870     
871               /* get the distance between the two leftmost extremes of lines
872                L1(0,1)<->L2(0,1) */
873               dx = lp0 - np0;
874               dy = lp1 - np1;
875               endPointDis = sqrt( dx * dx + dy * dy );
876     
877               /* set momentaneously min and max distance between lines to
878                the one between left extremes */
879               minLocalDis = endPointDis;
880               maxLocalDis = endPointDis;
881     
882               /* compute distance between right extremes
883                L1(2,3)<->L2(2,3) */
884               dx = lp2 - np2;
885               dy = lp3 - np3;
886               endPointDis = sqrt( dx * dx + dy * dy );
887     
888               /* update (if necessary) min and max distance between lines */
889               minLocalDis = ( endPointDis < minLocalDis ) ? endPointDis : minLocalDis;
890               maxLocalDis = ( endPointDis > maxLocalDis ) ? endPointDis : maxLocalDis;
891     
892               /* compute distance between left extreme of current line and
893                right extreme of line to be compared
894                L1(0,1)<->L2(2,3) */
895               dx = lp0 - np2;
896               dy = lp1 - np3;
897               endPointDis = sqrt( dx * dx + dy * dy );
898     
899               /* update (if necessary) min and max distance between lines */
900               minLocalDis = ( endPointDis < minLocalDis ) ? endPointDis : minLocalDis;
901               maxLocalDis = ( endPointDis > maxLocalDis ) ? endPointDis : maxLocalDis;
902     
903               /* compute distance between right extreme of current line and
904                left extreme of line to be compared
905                L1(2,3)<->L2(0,1) */
906               dx = lp2 - np0;
907               dy = lp3 - np1;
908               endPointDis = sqrt( dx * dx + dy * dy );
909     
910               /* update (if necessary) min and max distance between lines */
911               minLocalDis = ( endPointDis < minLocalDis ) ? endPointDis : minLocalDis;
912               maxLocalDis = ( endPointDis > maxLocalDis ) ? endPointDis : maxLocalDis;
913     
914               /* check whether conditions for considering line to be compared
915                worth to be inserted in the same LineVec are satisfied */
916               if( ( maxLocalDis < 0.8 * ( length + octaveLines[lineNextId].lineLength ) ) && ( minLocalDis < minEndPointDis ) )
917               {  //keep the closest line
918                 minEndPointDis = minLocalDis;
919                 closeLineID = lineNextId;
920               }
921             }
922     
923             /* add current line into octaveLines */
924             if( minEndPointDis < 12 )
925             {
926               octaveLines[numOfFinalLine].lineIDInScaleLineVec = octaveLines[closeLineID].lineIDInScaleLineVec;
927             }
928             else
929             {
930               octaveLines[numOfFinalLine].lineIDInScaleLineVec = lineIDInScaleLineVec;
931               lineIDInScaleLineVec++;
932             }
933             octaveLines[numOfFinalLine].octaveCount = octaveCount;
934             octaveLines[numOfFinalLine].lineIDInOctave = lineCurId;
935             octaveLines[numOfFinalLine].lineLength = length;
936             numOfFinalLine++;
937           }
938         }  //end for(unsigned int octaveCount = 1; octaveCount<numOfOctave_; octaveCount++)
939       }  //end if(numOfOctave_>1)
940     
941       ////////////////////////////////////
942       //Reorganize the detected lines into keyLines
943       keyLines.clear();
944       keyLines.resize( lineIDInScaleLineVec );
945       unsigned int tempID;
946       float s1, e1, s2, e2;
947       bool shouldChange;
948       OctaveSingleLine singleLine;
949       for ( unsigned int lineID = 0; lineID < numOfFinalLine; lineID++ )
950       {
951         lineIDInOctave = octaveLines[lineID].lineIDInOctave;
952         octaveID = octaveLines[lineID].octaveCount;
953         direction = edLineVec_[octaveID]->lineDirection_[lineIDInOctave];
954         singleLine.octaveCount = octaveID;
955         singleLine.direction = direction;
956         singleLine.lineLength = octaveLines[lineID].lineLength;
957         singleLine.salience = edLineVec_[octaveID]->lineSalience_[lineIDInOctave];
958         singleLine.numOfPixels = edLineVec_[octaveID]->lines_.sId[lineIDInOctave + 1] - edLineVec_[octaveID]->lines_.sId[lineIDInOctave];
959         //decide the start point and end point
960         shouldChange = false;
961         s1 = edLineVec_[octaveID]->lineEndpoints_[lineIDInOctave][0];  //sx
962         s2 = edLineVec_[octaveID]->lineEndpoints_[lineIDInOctave][1];  //sy
963         e1 = edLineVec_[octaveID]->lineEndpoints_[lineIDInOctave][2];  //ex
964         e2 = edLineVec_[octaveID]->lineEndpoints_[lineIDInOctave][3];  //ey
965         dx = e1 - s1;  //ex-sx
966         dy = e2 - s2;  //ey-sy
967         if( direction >= -0.75 * M_PI && direction < -0.25 * M_PI )
968         {
969           if( dy > 0 )
970           {
971             shouldChange = true;
972           }
973         }
974         if( direction >= -0.25 * M_PI && direction < 0.25 * M_PI )
975         {
976           if( dx < 0 )
977           {
978             shouldChange = true;
979           }
980         }
981         if( direction >= 0.25 * M_PI && direction < 0.75 * M_PI )
982         {
983           if( dy < 0 )
984           {
985             shouldChange = true;
986           }
987         }
988         if( ( direction >= 0.75 * M_PI && direction < M_PI ) || ( direction >= -M_PI && direction < -0.75 * M_PI ) )
989         {
990           if( dx > 0 )
991           {
992             shouldChange = true;
993           }
994         }
995         tempValue = scale[octaveID];
996         if( shouldChange )
997         {
998           singleLine.sPointInOctaveX = e1;
999           singleLine.sPointInOctaveY = e2;
1000           singleLine.ePointInOctaveX = s1;
1001           singleLine.ePointInOctaveY = s2;
1002           singleLine.startPointX = tempValue * e1;
1003           singleLine.startPointY = tempValue * e2;
1004           singleLine.endPointX = tempValue * s1;
1005           singleLine.endPointY = tempValue * s2;
1006         }
1007         else
1008         {
1009           singleLine.sPointInOctaveX = s1;
1010           singleLine.sPointInOctaveY = s2;
1011           singleLine.ePointInOctaveX = e1;
1012           singleLine.ePointInOctaveY = e2;
1013           singleLine.startPointX = tempValue * s1;
1014           singleLine.startPointY = tempValue * s2;
1015           singleLine.endPointX = tempValue * e1;
1016           singleLine.endPointY = tempValue * e2;
1017         }
1018         tempID = octaveLines[lineID].lineIDInScaleLineVec;
1019         keyLines[tempID].push_back( singleLine );
1020       }
1021     
1022       delete[] scale;
1023       return 1;
1024     }

1051

1053       short lengthOfLSP;  //the length of line support region, varies with lines
1054       short halfHeight = ( heightOfLSP - 1 ) / 2;
1055       short halfWidth;
1056       short bandID;
1057       float coefInGaussion;
1058       float lineMiddlePointX, lineMiddlePointY;
1059       float sCorX, sCorY, sCorX0, sCorY0;

1069       short octaveCount;
1070       OctaveSingleLine *pSingleLine;
1071       /* loop over list of LineVec */
1072       for ( short lineIDInScaleVec = 0; lineIDInScaleVec < numOfFinalLine; lineIDInScaleVec++ )
1073       {
1074         sameLineSize = (short) ( keyLines[lineIDInScaleVec].size() );
1075         /* loop over current LineVec's lines */
1076         for ( short lineIDInSameLine = 0; lineIDInSameLine < sameLineSize; lineIDInSameLine++ )
1077         {
1078           /* get a line in current LineVec and its original ID in its octave */
1079           pSingleLine = & ( keyLines[lineIDInScaleVec][lineIDInSameLine] );
1080           octaveCount = (short) pSingleLine->octaveCount;
1081     
1082           if( useDetectionData )
1083           {
1084             /* retrieve associated dxImg and dyImg */
1085             pdxImg = edLineVec_[octaveCount]->dxImg_.ptr<short>();
1086             pdyImg = edLineVec_[octaveCount]->dyImg_.ptr<short>();
1087     
1088             /* get image size to work on from real one */
1089             realWidth = (short) edLineVec_[octaveCount]->imageWidth;
1090             imageWidth = realWidth - 1;
1091             imageHeight = (short) ( edLineVec_[octaveCount]->imageHeight - 1 );
1092           }
1093     
1094           else
1095           {
1096             /* retrieve associated dxImg and dyImg */
1097             pdxImg = dxImg_vector[octaveCount].ptr<short>();
1098             pdyImg = dyImg_vector[octaveCount].ptr<short>();
1099     
1100             /* get image size to work on from real one */
1101             realWidth = (short) images_sizes[octaveCount].width;
1102             imageWidth = realWidth - 1;
1103             imageHeight = (short) ( images_sizes[octaveCount].height - 1 );
1104           }
1105     
1106           /* initialize memory areas */
1107           memset( pgdLBandSum, 0, numOfBitsBand );
1108           memset( ngdLBandSum, 0, numOfBitsBand );
1109           memset( pgdL2BandSum, 0, numOfBitsBand );
1110           memset( ngdL2BandSum, 0, numOfBitsBand );
1111           memset( pgdOBandSum, 0, numOfBitsBand );
1112           memset( ngdOBandSum, 0, numOfBitsBand );
1113           memset( pgdO2BandSum, 0, numOfBitsBand );
1114           memset( ngdO2BandSum, 0, numOfBitsBand );
1115     
1116           /* get length of line and its half */
1117           lengthOfLSP = (short) keyLines[lineIDInScaleVec][lineIDInSameLine].numOfPixels;
1118           halfWidth = ( lengthOfLSP - 1 ) / 2;
1119     
1120           /* get middlepoint of line */
1121           lineMiddlePointX = (float) ( 0.5 * ( pSingleLine->sPointInOctaveX + pSingleLine->ePointInOctaveX ) );
1122           lineMiddlePointY = (float) ( 0.5 * ( pSingleLine->sPointInOctaveY + pSingleLine->ePointInOctaveY ) );
1123     
1124           /*1.rotate the local coordinate system to the line direction (direction is the angle
1125            between positive line direction and positive X axis)
1126            *2.compute the gradient projection of pixels in line support region*/
1127     
1128           /* get the vector representing original image reference system after rotation to aligh with
1129            line's direction */
1130           dL[0] = cos( pSingleLine->direction );
1131           dL[1] = sin( pSingleLine->direction );
1132     
1133           /* set the clockwise orthogonal vector of line direction */
1134           dO[0] = -dL[1];
1135           dO[1] = dL[0];
1136     
1137           /* get rotated reference frame */
1138           sCorX0 = -dL[0] * halfWidth + dL[1] * halfHeight + lineMiddlePointX;  //hID =0; wID = 0;
1139           sCorY0 = -dL[1] * halfWidth - dL[0] * halfHeight + lineMiddlePointY;
1140     
1141           /* BIAS::Matrix<float> gDLMat(heightOfLSP,lengthOfLSP) */
1142           for ( short hID = 0; hID < heightOfLSP; hID++ )
1143           {
1144             /*initialization */
1145             sCorX = sCorX0;
1146             sCorY = sCorY0;
1147     
1148             pgdLRowSum = 0;
1149             ngdLRowSum = 0;
1150             pgdORowSum = 0;
1151             ngdORowSum = 0;
1152     
1153             for ( short wID = 0; wID < lengthOfLSP; wID++ )
1154             {
1155               tempCor = (short) round( sCorX );
1156               xCor = ( tempCor < 0 ) ? 0 : ( tempCor > imageWidth ) ? imageWidth : tempCor;
1157               tempCor = (short) round( sCorY );
1158               yCor = ( tempCor < 0 ) ? 0 : ( tempCor > imageHeight ) ? imageHeight : tempCor;
1159     
1160               /* To achieve rotation invariance, each simple gradient is rotated aligned with
1161                * the line direction and clockwise orthogonal direction.*/
1162               dx = pdxImg[yCor * realWidth + xCor];
1163               dy = pdyImg[yCor * realWidth + xCor];
1164               gDL = dx * dL[0] + dy * dL[1];
1165               gDO = dx * dO[0] + dy * dO[1];
1166               if( gDL > 0 )
1167               {
1168                 pgdLRowSum += gDL;
1169               }
1170               else
1171               {
1172                 ngdLRowSum -= gDL;
1173               }
1174               if( gDO > 0 )
1175               {
1176                 pgdORowSum += gDO;
1177               }
1178               else
1179               {
1180                 ngdORowSum -= gDO;
1181               }
1182               sCorX += dL[0];
1183               sCorY += dL[1];
1184               /* gDLMat[hID][wID] = gDL; */
1185             }
1186             sCorX0 -= dL[1];
1187             sCorY0 += dL[0];
1188             coefInGaussion = (float) gaussCoefG_[hID];
1189             pgdLRowSum = coefInGaussion * pgdLRowSum;
1190             ngdLRowSum = coefInGaussion * ngdLRowSum;
1191             pgdL2RowSum = pgdLRowSum * pgdLRowSum;
1192             ngdL2RowSum = ngdLRowSum * ngdLRowSum;
1193             pgdORowSum = coefInGaussion * pgdORowSum;
1194             ngdORowSum = coefInGaussion * ngdORowSum;
1195             pgdO2RowSum = pgdORowSum * pgdORowSum;
1196             ngdO2RowSum = ngdORowSum * ngdORowSum;
1197     
1198             /* compute {g_dL |g_dL>0 }, {g_dL |g_dL<0 },
1199              {g_dO |g_dO>0 }, {g_dO |g_dO<0 } of each band in the line support region
1200              first, current row belong to current band */
1201             bandID = (short) ( hID / params.widthOfBand_ );
1202             coefInGaussion = (float) ( gaussCoefL_[hID % params.widthOfBand_ + params.widthOfBand_] );
1203             pgdLBandSum[bandID] += coefInGaussion * pgdLRowSum;
1204             ngdLBandSum[bandID] += coefInGaussion * ngdLRowSum;
1205             pgdL2BandSum[bandID] += coefInGaussion * coefInGaussion * pgdL2RowSum;
1206             ngdL2BandSum[bandID] += coefInGaussion * coefInGaussion * ngdL2RowSum;
1207             pgdOBandSum[bandID] += coefInGaussion * pgdORowSum;
1208             ngdOBandSum[bandID] += coefInGaussion * ngdORowSum;
1209             pgdO2BandSum[bandID] += coefInGaussion * coefInGaussion * pgdO2RowSum;
1210             ngdO2BandSum[bandID] += coefInGaussion * coefInGaussion * ngdO2RowSum;
1211     
1212             /* In order to reduce boundary effect along the line gradient direction,
1213              * a row's gradient will contribute not only to its current band, but also
1214              * to its nearest upper and down band with gaussCoefL_.*/
1215             bandID--;
1216             if( bandID >= 0 )
1217             {/* the band above the current band */
1218               coefInGaussion = (float) ( gaussCoefL_[hID % params.widthOfBand_ + 2 * params.widthOfBand_] );
1219               pgdLBandSum[bandID] += coefInGaussion * pgdLRowSum;
1220               ngdLBandSum[bandID] += coefInGaussion * ngdLRowSum;
1221               pgdL2BandSum[bandID] += coefInGaussion * coefInGaussion * pgdL2RowSum;
1222               ngdL2BandSum[bandID] += coefInGaussion * coefInGaussion * ngdL2RowSum;
1223               pgdOBandSum[bandID] += coefInGaussion * pgdORowSum;
1224               ngdOBandSum[bandID] += coefInGaussion * ngdORowSum;
1225               pgdO2BandSum[bandID] += coefInGaussion * coefInGaussion * pgdO2RowSum;
1226               ngdO2BandSum[bandID] += coefInGaussion * coefInGaussion * ngdO2RowSum;
1227             }
1228             bandID = bandID + 2;
1229             if( bandID < NUM_OF_BANDS )
1230             {/*the band below the current band */
1231               coefInGaussion = (float) ( gaussCoefL_[hID % params.widthOfBand_] );
1232               pgdLBandSum[bandID] += coefInGaussion * pgdLRowSum;
1233               ngdLBandSum[bandID] += coefInGaussion * ngdLRowSum;
1234               pgdL2BandSum[bandID] += coefInGaussion * coefInGaussion * pgdL2RowSum;
1235               ngdL2BandSum[bandID] += coefInGaussion * coefInGaussion * ngdL2RowSum;
1236               pgdOBandSum[bandID] += coefInGaussion * pgdORowSum;
1237               ngdOBandSum[bandID] += coefInGaussion * ngdORowSum;
1238               pgdO2BandSum[bandID] += coefInGaussion * coefInGaussion * pgdO2RowSum;
1239               ngdO2BandSum[bandID] += coefInGaussion * coefInGaussion * ngdO2RowSum;
1240             }
1241           }
1242           /* gDLMat.Save("gDLMat.txt");
1243            return 0; */
1244     
1245           /* construct line descriptor */
1246           pSingleLine->descriptor.resize( descriptor_size );
1247           desVec = &pSingleLine->descriptor.front();
1248     
1249           short desID;
1250     
1251           /*Note that the first and last bands only have (lengthOfLSP * widthOfBand_ * 2.0) pixels
1252            * which are counted. */
1253           float invN2 = (float) ( 1.0 / ( params.widthOfBand_ * 2.0 ) );
1254           float invN3 = (float) ( 1.0 / ( params.widthOfBand_ * 3.0 ) );
1255           float invN, temp;
1256           for ( bandID = 0; bandID < NUM_OF_BANDS; bandID++ )
1257           {
1258             if( bandID == 0 || bandID == NUM_OF_BANDS - 1 )
1259             {
1260               invN = invN2;
1261             }
1262             else
1263             {
1264               invN = invN3;
1265             }
1266             desID = bandID * 8;
1267             temp = pgdLBandSum[bandID] * invN;
1268             desVec[desID] = temp;/* mean value of pgdL; */
1269             desVec[desID + 4] = sqrt( pgdL2BandSum[bandID] * invN - temp * temp );  //std value of pgdL;
1270             temp = ngdLBandSum[bandID] * invN;
1271             desVec[desID + 1] = temp;  //mean value of ngdL;
1272             desVec[desID + 5] = sqrt( ngdL2BandSum[bandID] * invN - temp * temp );  //std value of ngdL;
1273     
1274             temp = pgdOBandSum[bandID] * invN;
1275             desVec[desID + 2] = temp;  //mean value of pgdO;
1276             desVec[desID + 6] = sqrt( pgdO2BandSum[bandID] * invN - temp * temp );  //std value of pgdO;
1277             temp = ngdOBandSum[bandID] * invN;
1278             desVec[desID + 3] = temp;  //mean value of ngdO;
1279             desVec[desID + 7] = sqrt( ngdO2BandSum[bandID] * invN - temp * temp );  //std value of ngdO;
1280           }
1281     
1282           // normalize;
1283           float tempM, tempS;
1284           tempM = 0;
1285           tempS = 0;
1286           desVec = &pSingleLine->descriptor.front();
1287     
1288           int base = 0;
1289           for ( short i = 0; i < (short) ( NUM_OF_BANDS * 8 ); ++base, i = (short) ( base * 8 ) )
1290           {
1291             tempM += * ( desVec + i ) * * ( desVec + i );  //desVec[8*i+0] * desVec[8*i+0];
1292             tempM += * ( desVec + i + 1 ) * * ( desVec + i + 1 );  //desVec[8*i+1] * desVec[8*i+1];
1293             tempM += * ( desVec + i + 2 ) * * ( desVec + i + 2 );  //desVec[8*i+2] * desVec[8*i+2];
1294             tempM += * ( desVec + i + 3 ) * * ( desVec + i + 3 );  //desVec[8*i+3] * desVec[8*i+3];
1295             tempS += * ( desVec + i + 4 ) * * ( desVec + i + 4 );  //desVec[8*i+4] * desVec[8*i+4];
1296             tempS += * ( desVec + i + 5 ) * * ( desVec + i + 5 );  //desVec[8*i+5] * desVec[8*i+5];
1297             tempS += * ( desVec + i + 6 ) * * ( desVec + i + 6 );  //desVec[8*i+6] * desVec[8*i+6];
1298             tempS += * ( desVec + i + 7 ) * * ( desVec + i + 7 );  //desVec[8*i+7] * desVec[8*i+7];
1299           }
1300     
1301           tempM = 1 / sqrt( tempM );
1302           tempS = 1 / sqrt( tempS );
1303           desVec = &pSingleLine->descriptor.front();
1304           base = 0;
1305           for ( short i = 0; i < (short) ( NUM_OF_BANDS * 8 ); ++base, i = (short) ( base * 8 ) )
1306           {
1307             * ( desVec + i ) = * ( desVec + i ) * tempM;  //desVec[8*i] =  desVec[8*i] * tempM;
1308             * ( desVec + 1 + i ) = * ( desVec + 1 + i ) * tempM;  //desVec[8*i+1] =  desVec[8*i+1] * tempM;
1309             * ( desVec + 2 + i ) = * ( desVec + 2 + i ) * tempM;  //desVec[8*i+2] =  desVec[8*i+2] * tempM;
1310             * ( desVec + 3 + i ) = * ( desVec + 3 + i ) * tempM;  //desVec[8*i+3] =  desVec[8*i+3] * tempM;
1311             * ( desVec + 4 + i ) = * ( desVec + 4 + i ) * tempS;  //desVec[8*i+4] =  desVec[8*i+4] * tempS;
1312             * ( desVec + 5 + i ) = * ( desVec + 5 + i ) * tempS;  //desVec[8*i+5] =  desVec[8*i+5] * tempS;
1313             * ( desVec + 6 + i ) = * ( desVec + 6 + i ) * tempS;  //desVec[8*i+6] =  desVec[8*i+6] * tempS;
1314             * ( desVec + 7 + i ) = * ( desVec + 7 + i ) * tempS;  //desVec[8*i+7] =  desVec[8*i+7] * tempS;
1315           }
1316     
1317           /* In order to reduce the influence of non-linear illumination,
1318            * a threshold is used to limit the value of element in the unit feature
1319            * vector no larger than this threshold. In Z.Wang's work, a value of 0.4 is found
1320            * empirically to be a proper threshold.*/
1321           desVec = &pSingleLine->descriptor.front();
1322           for ( short i = 0; i < descriptor_size; i++ )
1323           {
1324             if( desVec[i] > 0.4 )
1325             {
1326               desVec[i] = (float) 0.4;
1327             }
1328           }
1329     
1330           //re-normalize desVec;
1331           temp = 0;
1332           for ( short i = 0; i < descriptor_size; i++ )
1333           {
1334             temp += desVec[i] * desVec[i];
1335           }
1336     
1337           temp = 1 / sqrt( temp );
1338           for ( short i = 0; i < descriptor_size; i++ )
1339           {
1340             desVec[i] = desVec[i] * temp;
1341           }
1342         }/* end for(short lineIDInSameLine = 0; lineIDInSameLine<sameLineSize;
1343          lineIDInSameLine++) */
1344     
1345         cv::Mat appoggio = cv::Mat( 1, 32, CV_32FC1 );
1346         float* pointerToRow = appoggio.ptr<float>( 0 );
1347         for ( int g = 0; g < 32; g++ )
1348         {
1349           /* get LBD data */
1350           float* des_Vec = &keyLines[lineIDInScaleVec][0].descriptor.front();
1351           *pointerToRow = des_Vec[g];
1352           pointerToRow++;
1353     
1354         }
1355     
1356       }/* end for(short lineIDInScaleVec = 0;
1357        lineIDInScaleVec<numOfFinalLine; lineIDInScaleVec++) */
1358     
1359       delete[] dL;
1360       delete[] dO;
1361       delete[] pgdLBandSum;
1362       delete[] ngdLBandSum;
1363       delete[] pgdL2BandSum;
1364       delete[] ngdL2BandSum;
1365       delete[] pgdOBandSum;
1366       delete[] ngdOBandSum;
1367       delete[] pgdO2BandSum;
1368       delete[] ngdO2BandSum;
1369     
1370       return 1;
1371     
1372     }

1442     int BinaryDescriptor::EDLineDetector::EdgeDrawing( cv::Mat &image, EdgeChains &edgeChains )
1443     {
1444       imageWidth = image.cols;
1445       imageHeight = image.rows;
1446       unsigned int pixelNum = imageWidth * imageHeight;
1447     
1448       unsigned int edgePixelArraySize = pixelNum / 5;
1449       unsigned int maxNumOfEdge = edgePixelArraySize / 20;
1450       //compute dx, dy images
1451       if( gImg_.cols != (int) imageWidth || gImg_.rows != (int) imageHeight )
1452       {
1453         if( pFirstPartEdgeX_ != NULL )
1454         {
1455           delete[] pFirstPartEdgeX_;
1456           delete[] pFirstPartEdgeY_;
1457           delete[] pSecondPartEdgeX_;
1458           delete[] pSecondPartEdgeY_;
1459           delete[] pFirstPartEdgeS_;
1460           delete[] pSecondPartEdgeS_;
1461           delete[] pAnchorX_;
1462           delete[] pAnchorY_;
1463         }
1464     
1465         dxImg_.create( imageHeight, imageWidth, CV_16SC1 );
1466         dyImg_.create( imageHeight, imageWidth, CV_16SC1 );
1467         gImgWO_.create( imageHeight, imageWidth, CV_16SC1 );
1468         gImg_.create( imageHeight, imageWidth, CV_16SC1 );
1469         dirImg_.create( imageHeight, imageWidth, CV_8UC1 );
1470         edgeImage_.create( imageHeight, imageWidth, CV_8UC1 );
1471         pFirstPartEdgeX_ = new unsigned int[edgePixelArraySize];
1472         pFirstPartEdgeY_ = new unsigned int[edgePixelArraySize];
1473         pSecondPartEdgeX_ = new unsigned int[edgePixelArraySize];
1474         pSecondPartEdgeY_ = new unsigned int[edgePixelArraySize];
1475         pFirstPartEdgeS_ = new unsigned int[maxNumOfEdge];
1476         pSecondPartEdgeS_ = new unsigned int[maxNumOfEdge];
1477         pAnchorX_ = new unsigned int[edgePixelArraySize];
1478         pAnchorY_ = new unsigned int[edgePixelArraySize];
1479       }
1480       cv::Sobel( image, dxImg_, CV_16SC1, 1, 0, 3 );
1481       cv::Sobel( image, dyImg_, CV_16SC1, 0, 1, 3 );
1482     
1483       //compute gradient and direction images
1484       cv::Mat dxABS_m = cv::abs( dxImg_ );
1485       cv::Mat dyABS_m = cv::abs( dyImg_ );
1486       cv::Mat sumDxDy;
1487       cv::add( dyABS_m, dxABS_m, sumDxDy );
1488     
1489       cv::threshold( sumDxDy, gImg_, gradienThreshold_ + 1, 255, cv::THRESH_TOZERO );
1490       gImg_ = gImg_ / 4;
1491       gImgWO_ = sumDxDy / 4;
1492       cv::compare( dxABS_m, dyABS_m, dirImg_, cv::CMP_LT );
1493     
1494       short *pgImg = gImg_.ptr<short>();
1495       unsigned char *pdirImg = dirImg_.ptr();
1496     
1497       //extract the anchors in the gradient image, store into a vector
1498       memset( pAnchorX_, 0, edgePixelArraySize * sizeof(unsigned int) );  //initialization
1499       memset( pAnchorY_, 0, edgePixelArraySize * sizeof(unsigned int) );
1500       unsigned int anchorsSize = 0;
1501       int indexInArray;
1502       unsigned char gValue1, gValue2, gValue3;
1503       for ( unsigned int w = 1; w < imageWidth - 1; w = w + scanIntervals_ )
1504       {
1505         for ( unsigned int h = 1; h < imageHeight - 1; h = h + scanIntervals_ )
1506         {
1507           indexInArray = h * imageWidth + w;
1508           //gValue1 = pdirImg[indexInArray];
1509           if( pdirImg[indexInArray] == Horizontal )
1510           {  //if the direction of pixel is horizontal, then compare with up and down
1511              //gValue2 = pgImg[indexInArray];
1512             if( pgImg[indexInArray] >= pgImg[indexInArray - imageWidth] + anchorThreshold_
1513                 && pgImg[indexInArray] >= pgImg[indexInArray + imageWidth] + anchorThreshold_ )
1514             {       // (w,h) is accepted as an anchor
1515               pAnchorX_[anchorsSize] = w;
1516               pAnchorY_[anchorsSize++] = h;
1517             }
1518           }
1519           else
1520           {       // if(pdirImg[indexInArray]==Vertical){//it is vertical edge, should be compared with left and right
1521             //gValue2 = pgImg[indexInArray];
1522             if( pgImg[indexInArray] >= pgImg[indexInArray - 1] + anchorThreshold_ && pgImg[indexInArray] >= pgImg[indexInArray + 1] + anchorThreshold_ )
1523             {       // (w,h) is accepted as an anchor
1524               pAnchorX_[anchorsSize] = w;
1525               pAnchorY_[anchorsSize++] = h;
1526             }
1527           }
1528         }
1529       }
1530       if( anchorsSize > edgePixelArraySize )
1531       {
1532         std::cout << "anchor size is larger than its maximal size. anchorsSize=" << anchorsSize << ", maximal size = " << edgePixelArraySize << std::endl;
1533         return -1;
1534       }
1535     
1536       //link the anchors by smart routing
1537       edgeImage_.setTo( 0 );
1538       unsigned char *pEdgeImg = edgeImage_.data;
1539       memset( pFirstPartEdgeX_, 0, edgePixelArraySize * sizeof(unsigned int) );       //initialization
1540       memset( pFirstPartEdgeY_, 0, edgePixelArraySize * sizeof(unsigned int) );
1541       memset( pSecondPartEdgeX_, 0, edgePixelArraySize * sizeof(unsigned int) );
1542       memset( pSecondPartEdgeY_, 0, edgePixelArraySize * sizeof(unsigned int) );
1543       memset( pFirstPartEdgeS_, 0, maxNumOfEdge * sizeof(unsigned int) );
1544       memset( pSecondPartEdgeS_, 0, maxNumOfEdge * sizeof(unsigned int) );
1545       unsigned int offsetPFirst = 0, offsetPSecond = 0;
1546       unsigned int offsetPS = 0;
1547     
1548       unsigned int x, y;
1549       unsigned int lastX = 0;
1550       unsigned int lastY = 0;
1551       unsigned char lastDirection;        //up = 1, right = 2, down = 3, left = 4;
1552       unsigned char shouldGoDirection;        //up = 1, right = 2, down = 3, left = 4;
1553       int edgeLenFirst, edgeLenSecond;
1554       for ( unsigned int i = 0; i < anchorsSize; i++ )
1555       {
1556         x = pAnchorX_[i];
1557         y = pAnchorY_[i];
1558         indexInArray = y * imageWidth + x;
1559         if( pEdgeImg[indexInArray] )
1560         {       //if anchor i is already been an edge pixel.
1561           continue;
1562         }
1563         /*The walk stops under 3 conditions:
1564          * 1. We move out of the edge areas, i.e., the thresholded gradient value
1565          *    of the current pixel is 0.
1566          * 2. The current direction of the edge changes, i.e., from horizontal
1567          *    to vertical or vice versa.?? (This is turned out not correct. From the online edge draw demo
1568          *    we can figure out that authors don't implement this rule either because their extracted edge
1569          *    chain could be a circle which means pixel directions would definitely be different
1570          *    in somewhere on the chain.)
1571          * 3. We encounter a previously detected edge pixel. */
1572         pFirstPartEdgeS_[offsetPS] = offsetPFirst;
1573         if( pdirImg[indexInArray] == Horizontal )
1574         {       //if the direction of this pixel is horizontal, then go left and right.
1575           //fist go right, pixel direction may be different during linking.
1576           lastDirection = RightDir;
1577           while ( pgImg[indexInArray] > 0 && !pEdgeImg[indexInArray] )
1578           {
1579             pEdgeImg[indexInArray] = 1;        // Mark this pixel as an edge pixel
1580             pFirstPartEdgeX_[offsetPFirst] = x;
1581             pFirstPartEdgeY_[offsetPFirst++] = y;
1582             shouldGoDirection = 0;        //unknown
1583             if( pdirImg[indexInArray] == Horizontal )
1584             {        //should go left or right
1585               if( lastDirection == UpDir || lastDirection == DownDir )
1586               {        //change the pixel direction now
1587                 if( x > lastX )
1588                 {        //should go right
1589                   shouldGoDirection = RightDir;
1590                 }
1591                 else
1592                 {        //should go left
1593                   shouldGoDirection = LeftDir;
1594                 }
1595               }
1596               lastX = x;
1597               lastY = y;
1598               if( lastDirection == RightDir || shouldGoDirection == RightDir )
1599               {        //go right
1600                 if( x == imageWidth - 1 || y == 0 || y == imageHeight - 1 )
1601                 {        //reach the image border
1602                   break;
1603                 }
1604                 // Look at 3 neighbors to the right and pick the one with the max. gradient value
1605                 gValue1 = (unsigned char) pgImg[indexInArray - imageWidth + 1];
1606                 gValue2 = (unsigned char) pgImg[indexInArray + 1];
1607                 gValue3 = (unsigned char) pgImg[indexInArray + imageWidth + 1];
1608                 if( gValue1 >= gValue2 && gValue1 >= gValue3 )
1609                 {        //up-right
1610                   x = x + 1;
1611                   y = y - 1;
1612                 }
1613                 else if( gValue3 >= gValue2 && gValue3 >= gValue1 )
1614                 {        //down-right
1615                   x = x + 1;

2646                                                             std::vector<double> &lineEquation, float &direction )
2647     {
2648       if( bValidate_ )
2649       {
2650         int n = offsetE - offsetS;
2651         /*first compute the direction of line, make sure that the dark side always be the
2652          *left side of a line.*/
2653         int meanGradientX = 0, meanGradientY = 0;
2654         short *pdxImg = dxImg_.ptr<short>();
2655         short *pdyImg = dyImg_.ptr<short>();

2657         std::vector<double> pointDirection;
2658         int index;
2659         for ( int i = 0; i < n; i++ )
2660         {
2661           index = yCors[offsetS] * imageWidth + xCors[offsetS];
2662           offsetS++;
2663           meanGradientX += pdxImg[index];
2664           meanGradientY += pdyImg[index];
2665           dx = (double) pdxImg[index];
2666           dy = (double) pdyImg[index];
2667           pointDirection.push_back( atan2( -dx, dy ) );
2668         }
2669         dx = fabs( lineEquation[1] );
2670         dy = fabs( lineEquation[0] );

2676         {         //first quadrant, and positive direction of X axis.
2677           direction = (float) atan2( -dy, dx );         //line direction is in fourth quadrant
2678         }
2679         if( meanGradientX <= 0 && meanGradientY > 0 )
2680         {         //second quadrant, and positive direction of Y axis.
2681           direction = (float) atan2( dy, dx );          //line direction is in first quadrant
2682         }
2683         if( meanGradientX < 0 && meanGradientY <= 0 )
2684         {         //third quadrant, and negative direction of X axis.
2685           direction = (float) atan2( dy, -dx );         //line direction is in second quadrant
2686         }
2687         if( meanGradientX >= 0 && meanGradientY < 0 )
2688         {         //fourth quadrant, and negative direction of Y axis.
2689           direction = (float) atan2( -dy, -dx );          //line direction is in third quadrant
2690         }
2691         /*then check whether the line is on the border of the image. We don't keep the border line.*/
2692         if( fabs( direction ) < 0.15 || M_PI - fabs( direction ) < 0.15 )
2693         {         //Horizontal line
2694           if( fabs( lineEquation[2] ) < 10 || fabs( imageHeight - fabs( lineEquation[2] ) ) < 10 )
2695           {         //upper border or lower border
2696             return false;
2697           }
2698         }
2699         if( fabs( fabs( direction ) - M_PI * 0.5 ) < 0.15 )
2700         {         //Vertical line
2701           if( fabs( lineEquation[2] ) < 10 || fabs( imageWidth - fabs( lineEquation[2] ) ) < 10 )
2702           {         //left border or right border
2703             return false;
2704           }
2705         }
2706         //count the aligned points on the line which have the same direction as the line.
2707         double disDirection;
2708         int k = 0;
2709         for ( int i = 0; i < n; i++ )
2710         {
2711           disDirection = fabs( direction - pointDirection[i] );
2712           if( fabs( 2 * M_PI - disDirection ) < 0.392699 || disDirection < 0.392699 )
2713           {         //same direction, pi/8 = 0.392699081698724
2714             k++;
2715           }
2716         }
2717         //now compute NFA(Number of False Alarms)
2718         double ret = nfa( n, k, 0.125, logNT_ );
2719     
2720         return ( ret > 0 );  //0 corresponds to 1 mean false alarm
2721       }
2722       else
2723       {
2724         return true;
2725       }
2726     }
