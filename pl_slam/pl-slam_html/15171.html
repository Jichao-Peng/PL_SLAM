83     /** @brief A class to represent a line
84     
85     As aformentioned, it is been necessary to design a class that fully stores the information needed to
86     characterize completely a line and plot it on image it was extracted from, when required.
87     
88     *KeyLine* class has been created for such goal; it is mainly inspired to Feature2d's KeyPoint class,
89     since KeyLine shares some of *KeyPoint*'s fields, even if a part of them assumes a different
90     meaning, when speaking about lines. In particular:
91     
92     -   the *class_id* field is used to gather lines extracted from different octaves which refer to
93         same line inside original image (such lines and the one they represent in original image share
94         the same *class_id* value)
95     -   the *angle* field represents line's slope with respect to (positive) X axis
96     -   the *pt* field represents line's midpoint
97     -   the *response* field is computed as the ratio between the line's length and maximum between
98         image's width and height
99     -   the *size* field is the area of the smallest rectangle containing line
100     
101     Apart from fields inspired to KeyPoint class, KeyLines stores information about extremes of line in
102     original image and in octave it was extracted from, about line's length and number of pixels it
103     covers.
104      */
105     struct CV_EXPORTS KeyLine
106     {
107      public:
108       /** orientation of the line */
109       float angle;
110     
111       /** object ID, that can be used to cluster keylines by the line they represent */
112       int class_id;
113     
114       /** octave (pyramid layer), from which the keyline has been extracted */
115       int octave;
116     
117       /** coordinates of the middlepoint */
118       Point2f pt;
119     
120       /** the response, by which the strongest keylines have been selected.
121        It's represented by the ratio between line's length and maximum between
122        image's width and height */
123       float response;
124     
125       /** minimum area containing line */
126       float size;
127     
128       /** lines's extremes in original image */
129       float startPointX;
130       float startPointY;
131       float endPointX;
132       float endPointY;
133     
134       /** line's extremes in image it was extracted from */
135       float sPointInOctaveX;
136       float sPointInOctaveY;
137       float ePointInOctaveX;
138       float ePointInOctaveY;
139     
140       /** the length of line */
141       float lineLength;
142     
143       /** number of pixels covered by the line */
144       int numOfPixels;
145     
146       /** Returns the start point of the line in the original image */
147       Point2f getStartPoint() const
148       {
149         return Point2f(startPointX, startPointY);
150       }
151     
152       /** Returns the end point of the line in the original image */
153       Point2f getEndPoint() const
154       {
155         return Point2f(endPointX, endPointY);
156       }
157     
158       /** Returns the start point of the line in the octave it was extracted from */
159       Point2f getStartPointInOctave() const
160       {
161         return Point2f(sPointInOctaveX, sPointInOctaveY);
162       }
163     
164       /** Returns the end point of the line in the octave it was extracted from */
165       Point2f getEndPointInOctave() const
166       {
167         return Point2f(ePointInOctaveX, ePointInOctaveY);
168       }
169     
170       /** constructor */
171       KeyLine()
172       {
173       }
174     };

622         /** Compare doubles by relative error.
623          The resulting rounding error after floating point computations
624          depend on the specific operations done. The same number computed by
625          different algorithms could present different rounding errors. For a
626          useful comparison, an estimation of the relative rounding error
627          should be considered and compared to a factor times EPS. The factor
628          should be related to the accumulated rounding error in the chain of
629          computation. Here, as a simplification, a fixed factor is used.
630          */
631         static int double_equal( double a, double b )
632         {
633           double abs_diff, aa, bb, abs_max;
634           /* trivial case */
635           if( a == b )
636             return true;
637           abs_diff = fabs( a - b );
638           aa = fabs( a );
639           bb = fabs( b );
640           abs_max = aa > bb ? aa : bb;
641     
642           /* DBL_MIN is the smallest normalized number, thus, the smallest
643            number whose relative error is bounded by DBL_EPSILON. For
644            smaller numbers, the same quantization steps as for DBL_MIN
645            are used. Then, for smaller numbers, a meaningful "relative"
646            error should be computed by dividing the difference by DBL_MIN. */
647           if( abs_max < DBL_MIN )
648             abs_max = DBL_MIN;
649     
650           /* equal if relative error <= factor x eps */
651           return ( abs_diff / abs_max ) <= ( RELATIVE_ERROR_FACTOR * DBL_EPSILON );
652         }
653     
654         /** Computes the natural logarithm of the absolute value of
655          the gamma function of x using the Lanczos approximation.
656          See http://www.rskey.org/gamma.htm
657          The formula used is
658          @f[
659          \Gamma(x) = \frac{ \sum_{n=0}^{N} q_n x^n }{ \Pi_{n=0}^{N} (x+n) }
660          (x+5.5)^{x+0.5} e^{-(x+5.5)}
661          @f]
662          so
663          @f[
664          \log\Gamma(x) = \log\left( \sum_{n=0}^{N} q_n x^n \right)
665          + (x+0.5) \log(x+5.5) - (x+5.5) - \sum_{n=0}^{N} \log(x+n)
666          @f]
667          and
668          q0 = 75122.6331530,
669          q1 = 80916.6278952,
670          q2 = 36308.2951477,
671          q3 = 8687.24529705,
672          q4 = 1168.92649479,
673          q5 = 83.8676043424,
674          q6 = 2.50662827511.
675          */
676         static double log_gamma_lanczos( double x )
677         {
678           static double q[7] =
679           { 75122.6331530, 80916.6278952, 36308.2951477, 8687.24529705, 1168.92649479, 83.8676043424, 2.50662827511 };
680           double a = ( x + 0.5 ) * log( x + 5.5 ) - ( x + 5.5 );
681           double b = 0.0;
682           int n;
683           for ( n = 0; n < 7; n++ )
684           {
685             a -= log( x + (double) n );
686             b += q[n] * pow( x, (double) n );
687           }
688           return a + log( b );
689         }
690     
691         /** Computes the natural logarithm of the absolute value of
692          the gamma function of x using Windschitl method.
693          See http://www.rskey.org/gamma.htm
694          The formula used is
695          @f[
696          \Gamma(x) = \sqrt{\frac{2\pi}{x}} \left( \frac{x}{e}
697          \sqrt{ x\sinh(1/x) + \frac{1}{810x^6} } \right)^x
698          @f]
699          so
700          @f[
701          \log\Gamma(x) = 0.5\log(2\pi) + (x-0.5)\log(x) - x
702          + 0.5x\log\left( x\sinh(1/x) + \frac{1}{810x^6} \right).
703          @f]
704          This formula is a good approximation when x > 15.
705          */
706         static double log_gamma_windschitl( double x )
707         {
708           return 0.918938533204673 + ( x - 0.5 ) * log( x ) - x + 0.5 * x * log( x * sinh( 1 / x ) + 1 / ( 810.0 * pow( x, 6.0 ) ) );
709         }

711         /** Computes -log10(NFA).
712          NFA stands for Number of False Alarms:
713          @f[
714          \mathrm{NFA} = NT \cdot B(n,k,p)
715          @f]
716          - NT       - number of tests
717          - B(n,k,p) - tail of binomial distribution with parameters n,k and p:
718          @f[
719          B(n,k,p) = \sum_{j=k}^n
720          \left(\begin{array}{c}n\\j\end{array}\right)
721          p^{j} (1-p)^{n-j}
722          @f]
723          The value -log10(NFA) is equivalent but more intuitive than NFA:
724          - -1 corresponds to 10 mean false alarms
725          -  0 corresponds to 1 mean false alarm
726          -  1 corresponds to 0.1 mean false alarms
727          -  2 corresponds to 0.01 mean false alarms
728          -  ...
729          Used this way, the bigger the value, better the detection,
730          and a logarithmic scale is used.
731          @param n,k,p binomial parameters.
732          @param logNT logarithm of Number of Tests
733          The computation is based in the gamma function by the following
734          relation:
735          @f[
736          \left(\begin{array}{c}n\\k\end{array}\right)
737          = \frac{ \Gamma(n+1) }{ \Gamma(k+1) \cdot \Gamma(n-k+1) }.
738          @f]
739          We use efficient algorithms to compute the logarithm of
740          the gamma function.
741          To make the computation faster, not all the sum is computed, part
742          of the terms are neglected based on a bound to the error obtained
743          (an error of 10% in the result is accepted).
744          */
745         static double nfa( int n, int k, double p, double logNT )
746         {
747           double tolerance = 0.1; /* an error of 10% in the result is accepted */
748           double log1term, term, bin_term, mult_term, bin_tail, err, p_term;
749           int i;
750     
751           /* check parameters */
752           if( n < 0 || k < 0 || k > n || p <= 0.0 || p >= 1.0 )
753           {
754             std::cout << "nfa: wrong n, k or p values." << std::endl;
755             exit( 0 );
756           }
757           /* trivial cases */
758           if( n == 0 || k == 0 )
759             return -logNT;
760           if( n == k )
761             return -logNT - (double) n * log10( p );
762     
763           /* probability term */
764           p_term = p / ( 1.0 - p );
765     
766           /* compute the first term of the series */
767           /*
768            binomial_tail(n,k,p) = sum_{i=k}^n bincoef(n,i) * p^i * (1-p)^{n-i}
769            where bincoef(n,i) are the binomial coefficients.
770            But
771            bincoef(n,k) = gamma(n+1) / ( gamma(k+1) * gamma(n-k+1) ).
772            We use this to compute the first term. Actually the log of it.
773            */
774           log1term = log_gamma( (double) n + 1.0 )- log_gamma( (double ) k + 1.0 )- log_gamma( (double ) ( n - k ) + 1.0 )
775     + (double) k * log( p )
776     + (double) ( n - k ) * log( 1.0 - p );
777     term = exp( log1term );
778     
779     /* in some cases no more computations are needed */
780     if( double_equal( term, 0.0 ) )
781     { /* the first term is almost zero */
782       if( (double) k > (double) n * p ) /* at begin or end of the tail?  */
783       return -log1term / MLN10 - logNT; /* end: use just the first term  */
784       else
785       return -logNT; /* begin: the tail is roughly 1  */
786     }
787     
788     /* compute more terms if needed */
789     bin_tail = term;
790     for ( i = k + 1; i <= n; i++ )
791     {
792       /*    As
793        term_i = bincoef(n,i) * p^i * (1-p)^(n-i)
794        and
795        bincoef(n,i)/bincoef(n,i-1) = n-i+1 / i,
796        then,
797        term_i / term_i-1 = (n-i+1)/i * p/(1-p)
798        and
799        term_i = term_i-1 * (n-i+1)/i * p/(1-p).
800        p/(1-p) is computed only once and stored in 'p_term'.
801        */
802       bin_term = (double) ( n - i + 1 ) / (double) i;
803       mult_term = bin_term * p_term;
804       term *= mult_term;
805       bin_tail += term;
806       if( bin_term < 1.0 )
807       {
808         /* When bin_term<1 then mult_term_j<mult_term_i for j>i.
809          Then, the error on the binomial tail when truncated at
810          the i term can be bounded by a geometric series of form
811          term_i * sum mult_term_i^j.                            */
812         err = term * ( ( 1.0 - pow( mult_term, (double) ( n - i + 1 ) ) ) / ( 1.0 - mult_term ) - 1.0 );
813         /* One wants an error at most of tolerance*final_result, or:
814          tolerance * abs(-log10(bin_tail)-logNT).
815          Now, the error that can be accepted on bin_tail is
816          given by tolerance*final_result divided by the derivative
817          of -log10(x) when x=bin_tail. that is:
818          tolerance * abs(-log10(bin_tail)-logNT) / (1/bin_tail)
819          Finally, we truncate the tail if the error is less than:
820          tolerance * abs(-log10(bin_tail)-logNT) * bin_tail        */
821         if( err < tolerance * fabs( -log10( bin_tail ) - logNT ) * bin_tail )
822         break;
823       }
824     }
825     return -log10( bin_tail ) - logNT;
826     }
827     };

959     /** @brief furnishes all functionalities for querying a dataset provided by user or internal to
960     class (that user must, anyway, populate) on the model of @ref features2d_match
961     
962     
963     Once descriptors have been extracted from an image (both they represent lines and points), it
964     becomes interesting to be able to match a descriptor with another one extracted from a different
965     image and representing the same line or point, seen from a differente perspective or on a different
966     scale. In reaching such goal, the main headache is designing an efficient search algorithm to
967     associate a query descriptor to one extracted from a dataset. In the following, a matching modality
968     based on *Multi-Index Hashing (MiHashing)* will be described.
969     
970     Multi-Index Hashing
971     -------------------
972     
973     The theory described in this section is based on @cite MIH . Given a dataset populated with binary
974     codes, each code is indexed *m* times into *m* different hash tables, according to *m* substrings it
975     has been divided into. Thus, given a query code, all the entries close to it at least in one
976     substring are returned by search as *neighbor candidates*. Returned entries are then checked for
977     validity by verifying that their full codes are not distant (in Hamming space) more than *r* bits
978     from query code. In details, each binary code **h** composed of *b* bits is divided into *m*
979     disjoint substrings \f$\mathbf{h}^{(1)}, ..., \mathbf{h}^{(m)}\f$, each with length
980     \f$\lfloor b/m \rfloor\f$ or \f$\lceil b/m \rceil\f$ bits. Formally, when two codes **h** and **g** differ
981     by at the most *r* bits, in at the least one of their *m* substrings they differ by at the most
982     \f$\lfloor r/m \rfloor\f$ bits. In particular, when \f$||\mathbf{h}-\mathbf{g}||_H \le r\f$ (where \f$||.||_H\f$
983     is the Hamming norm), there must exist a substring *k* (with \f$1 \le k \le m\f$) such that
984     
985     \f[||\mathbf{h}^{(k)} - \mathbf{g}^{(k)}||_H \le \left\lfloor \frac{r}{m} \right\rfloor .\f]
986     
987     That means that if Hamming distance between each of the *m* substring is strictly greater than
988     \f$\lfloor r/m \rfloor\f$, then \f$||\mathbf{h}-\mathbf{g}||_H\f$ must be larger that *r* and that is a
989     contradiction. If the codes in dataset are divided into *m* substrings, then *m* tables will be
990     built. Given a query **q** with substrings \f$\{\mathbf{q}^{(i)}\}^m_{i=1}\f$, *i*-th hash table is
991     searched for entries distant at the most \f$\lfloor r/m \rfloor\f$ from \f$\mathbf{q}^{(i)}\f$ and a set of
992     candidates \f$\mathcal{N}_i(\mathbf{q})\f$ is obtained. The union of sets
993     \f$\mathcal{N}(\mathbf{q}) = \bigcup_i \mathcal{N}_i(\mathbf{q})\f$ is a superset of the *r*-neighbors
994     of **q**. Then, last step of algorithm is computing the Hamming distance between **q** and each
995     element in \f$\mathcal{N}(\mathbf{q})\f$, deleting the codes that are distant more that *r* from **q**.
996     */
