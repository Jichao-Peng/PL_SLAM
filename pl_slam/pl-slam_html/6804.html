93     		/** Reads a configuration parameter of type vector, stored in the file as a string: "[v1 v2 v3 ... ]", where spaces could also be commas. \exception std::exception If the key name is not found and "failIfNotFound" is true. Otherwise the "defaultValue" is returned. */
94     		template <class VECTOR_TYPE>
95     		void  read_vector(
96     			const std::string  & section,
97     			const std::string  & name,
98     			const VECTOR_TYPE  & defaultValue,
99     			VECTOR_TYPE        & outValues,
100     			bool                 failIfNotFound = false) const
101     		{
102     			std::string aux ( readString(section, name, "",failIfNotFound ) );
103     			// Parse the text into a vector:
104     			std::vector<std::string>	tokens;
105     			mrpt::system::tokenize( aux,"[], \t",tokens);
106     
107     			if (tokens.size()==0)
108     			{
109     				outValues = defaultValue;
110     			}
111     			else
112     			{
113     				// Parse to numeric type:
114     				const size_t N = tokens.size();
115     				outValues.resize( N );
116     				for (size_t i=0;i<N;i++)
117     				{
118     					std::stringstream ss(tokens[i]);
119     					ss >> outValues[i];
120     				}
121     			}
122     		}

124     		/** Reads a configuration parameter as a matrix written in a matlab-like format - for example: "[2 3 4 ; 7 8 9]".
125     		 * This template method can be instantiated for matrices of the types: int, long, unsinged int, unsigned long, float, double, long double
126     		 * \exception std::exception If the key name is not found and "failIfNotFound" is true. Otherwise the "defaultValue" is returned.
127     		 */
128     		 template <class MATRIX_TYPE>
129     		 void read_matrix(
130     			const std::string			&section,
131     			const std::string			&name,
132     			MATRIX_TYPE	&outMatrix,
133     			const MATRIX_TYPE &defaultMatrix = MATRIX_TYPE(),
134     			bool failIfNotFound = false ) const
135     		{
136     			std::string aux = readString(section, name, "",failIfNotFound );
137     			if (aux.empty())
138     				outMatrix = defaultMatrix;
139     			else
140     			{
141     				// Parse the text into a vector:
142     				if (!outMatrix.fromMatlabStringFormat(aux))
143     					THROW_EXCEPTION_CUSTOM_MSG1("Error parsing matrix: '%s'",aux.c_str())
144     			}
145     		}

147     		/** Reads an "enum" value, where the value in the config file can be either a numerical value or the symbolic name, for example:
148     		  *   In the code:
149     		  *  \code
150     		  *    enum my_type_t { type_foo=0, type_bar };
151     		  *  \endcode
152     		  *  In the config file:
153     		  *  \code
154     		  *    [section]
155     		  *    type   = type_bar   // Use the symbolic name, or
156     		  *    type   = 1          // use the numerical value (both lines will be equivalent)
157     		  *  \endcode
158     		  *  Which can be loaded with:
159     		  *  \code
160     		  *    cfgfile.read_enum<my_type_t>("section","type", type_foo );
161     		  *  \endcode
162     		  *
163     		  *  \note For an enum type to work with this template it is required that it defines a specialization of mrpt::utils::TEnumType
164     		  */
165     		template <typename ENUMTYPE>
166     		ENUMTYPE read_enum(const std::string &section, const std::string &name, const ENUMTYPE &defaultValue, bool failIfNotFound = false) const
167     		{
168     			MRPT_START
169     			const std::string sVal = read_string_first_word(section,name,"",failIfNotFound);
170     			if (sVal.empty()) return defaultValue;
171     			// Text or numeric value?
172     			if (::isdigit(sVal[0]))
173     			{	// Seems a number:
174     				return static_cast<ENUMTYPE>(::atoi(&sVal[0]));
175     			}
176     			else
177     			{	// Name look-up:
178     				try {
179     				return mrpt::utils::TEnumType<ENUMTYPE>::name2value(sVal);
180     				} catch (std::exception &)
181     				{
182     					THROW_EXCEPTION(format("Invalid value '%s' for enum type while reading key='%s'.",sVal.c_str(),name.c_str()))
183     				}
184     			}
185     			MRPT_END
186     		}
