291     		/** Projects any 3D object into the plane's base, using its inverse pose. If the object is exactly inside the plane, this projection will zero its Z coordinates */
292     		template<class T> void project3D(const T &obj,const TPlane &newXYPlane,T &newObj)	{
293     			mrpt::poses::CPose3D pose;
294     			TPlane(newXYPlane).getAsPose3D(pose);
295     			project3D(obj,-pose,newObj);
296     		}

298     		/** Projects any 3D object into the plane's base, using its inverse pose and forcing the position of the new coordinates origin. If the object is exactly inside the plane, this projection will zero its Z coordinates  */
299     		template<class T> void project3D(const T &obj,const TPlane &newXYPlane,const TPoint3D &newOrigin,T &newObj)	{
300     			mrpt::poses::CPose3D pose;
301     			//TPlane(newXYPlane).getAsPose3DForcingOrigin(newOrigin,pose);
302     			TPlane(newXYPlane).getAsPose3D(pose);
303     			project3D(obj,-pose,newObj);
304     		}

308     			size_t N=objs.size();

404     		/** Gets the intersection between vectors of geometric objects and returns it in a sparse matrix of either TObject2D or TObject3D.
405     		  * \sa TObject2D,TObject3D,CSparseMatrix */
406     		template<class T,class U,class O> size_t intersect(const std::vector<T> &v1,const std::vector<U> &v2,CSparseMatrixTemplate<O> &objs)	{
407     			size_t M=v1.size(),N=v2.size();
408     			O obj;
409     			objs.clear();
410     			objs.resize(M,N);
411     			for (size_t i=0;i<M;i++) for (size_t j=0;j<M;j++) if (intersect(v1[i],v2[j],obj)) objs(i,j)=obj;
412     			return objs.getNonNullElements();
413     		}

416     		  * \sa TObject2D,TObject3D */
417     		template<class T,class U,class O> size_t intersect(const std::vector<T> &v1,const std::vector<U> &v2,std::vector<O> objs)	{
418     			objs.resize(0);
419     			O obj;
420     			for (typename std::vector<T>::const_iterator it1=v1.begin();it1!=v1.end();++it1)	{
421     				const T &elem1=*it1;
422     				for (typename std::vector<U>::const_iterator it2=v2.begin();it2!=v2.end();++it2) if (intersect(elem1,*it2,obj)) objs.push_back(obj);
423     			}
424     			return objs.size();
425     		}

614     		/** Computes the cross product of two 3D vectors, returning a vector normal to both.
615     		  *  It uses the simple implementation:
616     
617     		    \f[  v_out = \left(
618     					\begin{array}{c c c}
619     					\hat{i} ~ \hat{j} ~ \hat{k} \\
620     					x0 ~ y0 ~ z0 \\
621     					x1 ~ y1 ~ z1 \\
622     					\end{array} \right)
623     			\f]
624     		  */
625     		template<class T,class U,class V>
626     		inline void crossProduct3D(const T &v0,const U &v1,V& vOut)	{
627     			vOut[0]=v0[1]*v1[2]-v0[2]*v1[1];
628     			vOut[1]=v0[2]*v1[0]-v0[0]*v1[2];
629     			vOut[2]=v0[0]*v1[1]-v0[1]*v1[0];
630     		}
631     
632     		//! \overload
633     		template<class T>
634     		inline void crossProduct3D(
635     			const std::vector<T> &v0,
636     			const std::vector<T> &v1,
637     			std::vector<T> &v_out )
638     		{
639     			ASSERT_(v0.size()==3)
640     			ASSERT_(v1.size()==3);
641     			v_out.resize(3);
642     			v_out[0] =  v0[1]*v1[2] - v0[2]*v1[1];
643     			v_out[1] = -v0[0]*v1[2] + v0[2]*v1[0];
644     			v_out[2] =  v0[0]*v1[1] - v0[1]*v1[0];
645     		}

648     		template<class VEC1,class VEC2>
649     		inline Eigen::Matrix<double,3,1> crossProduct3D(const VEC1 &v0,const VEC2 &v1)	{
650     			Eigen::Matrix<double,3,1> vOut;
651     			vOut[0]=v0[1]*v1[2]-v0[2]*v1[1];
652     			vOut[1]=v0[2]*v1[0]-v0[0]*v1[2];
653     			vOut[2]=v0[0]*v1[1]-v0[1]*v1[0];
654     			return vOut;

657     		/** Computes the 3x3 skew symmetric matrix from a 3-vector or 3-array:
658     		  * \f[  M([x ~ y ~ z]^\top) = \left(
659     		  * 	\begin{array}{c c c}
660     		  * 	0 & -z & y \\
661     		  * 	z & 0 & -x \\
662     		  * 	-y & x & 0
663     		  * 	\end{array} \right)
664     		  * \f]
665     		  */
666     		template<class VECTOR,class MATRIX>
667     		inline void skew_symmetric3(const VECTOR &v,MATRIX& M)	{
668     			ASSERT_(v.size()==3)
669     			M.setSize(3,3);
670     			M.set_unsafe(0,0, 0); M.set_unsafe(0,1, -v[2]); M.set_unsafe(0,2, v[1]);
671     			M.set_unsafe(1,0, v[2]); M.set_unsafe(1,1, 0); M.set_unsafe(1,2, -v[0]);
672     			M.set_unsafe(2,0, -v[1]); M.set_unsafe(2,1, v[0]); M.set_unsafe(2,2, 0);
673     		}
674     		//! \overload
675     		template<class VECTOR>
676     		inline mrpt::math::CMatrixDouble33 skew_symmetric3(const VECTOR &v)	{
677     			mrpt::math::CMatrixDouble33 M(mrpt::math::UNINITIALIZED_MATRIX);
678     			skew_symmetric3(v,M);
679     			return M;
680     		}

834     		/** Specialized method to check whether a point (x,y) falls into a quadrangle.
835     		  * \sa pointIntoPolygon2D
836     		  */
837     		template <typename T>
838     		bool pointIntoQuadrangle(
839     			T x, T y,
840     			T v1x, T v1y,
841     			T v2x, T v2y,
842     			T v3x, T v3y,
843     			T v4x, T v4y )
844     		{
845     			using mrpt::utils::sign;
846     
847     			const T a1 = atan2( v1y - y , v1x - x );
848     			const T a2 = atan2( v2y - y , v2x - x );
849     			const T a3 = atan2( v3y - y , v3x - x );
850     			const T a4 = atan2( v4y - y , v4x - x );
851     
852     			// The point is INSIDE iff all the signs of the angles between each vertex
853     			//  and the next one are equal.
854     			const T da1 = mrpt::math::wrapToPi( a2-a1 );
855     			const T da2 = mrpt::math::wrapToPi( a3-a2 );
856     			if (sign(da1)!=sign(da2)) return false;
857     
858     			const T da3 = mrpt::math::wrapToPi( a4-a3 );
859     			if (sign(da2)!=sign(da3)) return false;
860     
861     			const T da4 = mrpt::math::wrapToPi( a1-a4 );
862     			return (sign(da3)==sign(da4) && (sign(da4)==sign(da1)));
863     		}

899     		/** Computes an axis base (a set of three 3D normal vectors) with the given vector being the first of them.
900     		  * NOTE: Make sure of passing all floats or doubles and that the template of the receiving matrix is of the same type!
901     		  *
902     		  *  If   \f$ d = [ dx ~ dy ~ dz ] \f$ is the input vector, then this function returns a matrix \f$ M \f$ such as:
903     		  *
904     		    \f[  M = \left(
905     					\begin{array}{c c c}
906     					v^1_x ~ v^2_x ~ v^3_x \\
907     					v^1_y ~ v^2_y ~ v^3_y \\
908     					v^1_z ~ v^2_z ~ v^3_z
909     					\end{array} \right)
910     			\f]
911     		  *
912     		  *   And the three normal vectors are computed as:
913     
914     			  \f[ v^1 = \frac{d}{|d|}  \f]
915     
916     			  If (dx!=0 or dy!=0):
917     				\f[ v^2 = \frac{[-dy ~ dx ~ 0 ]}{\sqrt{dx^2+dy^2}}  \f]
918     			  otherwise (the direction vector is vertical):
919     				\f[ v^2 = [1 ~ 0 ~ 0]  \f]
920     
921     			  And finally, the third vector is the cross product of the others:
922     
923     			    \f[ v^3 = v^1 \times v^2  \f]
924     		  *
925     		  * \return The 3x3 matrix (CMatrixTemplateNumeric<T>), containing one vector per column.
926     		  * \except Throws an std::exception on invalid input (i.e. null direction vector)
927     		  *
928     		  * (JLB @ 18-SEP-2007)
929     		  */
930     		template<class T>
931     		CMatrixTemplateNumeric<T> generateAxisBaseFromDirection( T dx, T dy, T dz )
932     		{
933     			MRPT_START
934     
935     			if (dx==0 && dy==0 && dz==0)
936     				THROW_EXCEPTION("Invalid input: Direction vector is (0,0,0)!");
937     
938     			CMatrixTemplateNumeric<T>	P(3,3);
939     
940     			// 1st vector:
941     			T	n_xy = square(dx)+square(dy);
942     			T	n = sqrt(n_xy+square(dz));
943     			n_xy = sqrt(n_xy);
944     			P(0,0) = dx / n;
945     			P(1,0) = dy / n;
946     			P(2,0) = dz / n;
947     
948     			// 2nd perpendicular vector:
949     			if (fabs(dx)>1e-4 || fabs(dy)>1e-4)
950     			{
951     				P(0,1) = -dy / n_xy;
952     				P(1,1) = dx / n_xy;
953     				P(2,1) = 0;
954     			}
955     			else
956     			{
957     				// Any vector in the XY plane will work:
958     				P(0,1) = 1;
959     				P(1,1) = 0;
960     				P(2,1) = 0;
961     			}
962     
963     			// 3rd perpendicular vector: cross product of the two last vectors:
964     			P.col(2) = crossProduct3D(P.col(0),P.col(1));
965     
966     			return P;
967     			MRPT_END
968     		}

971     		/** Compute a rotation exponential using the Rodrigues Formula.
972     		  * The rotation axis is given by \f$\vec{w}\f$, and the rotation angle must
973     		  * be computed using \f$ \theta = |\vec{w}|\f$. This is provided as a separate
974     		  * function primarily to allow fast and rough matrix exponentials using fast
975     		  * and rough approximations to \e A and \e B.
976     		  *
977     		  * \param w Vector about which to rotate.
978     		  * \param A \f$\frac{\sin \theta}{\theta}\f$
979     		  * \param B \f$\frac{1 - \cos \theta}{\theta^2}\f$
980     		  * \param R Matrix to hold the return value.
981     		  * \sa CPose3D
982     		  * \note Method from TooN (C) Tom Drummond (GNU GPL)
983     		  */
984     		template <typename VECTOR_LIKE, typename Precision, typename MATRIX_LIKE>
985     		inline void rodrigues_so3_exp(const VECTOR_LIKE& w, const Precision A,const Precision B,MATRIX_LIKE & R)
986     		{
987     			ASSERT_EQUAL_(w.size(),3)
988     			ASSERT_EQUAL_(R.getColCount(),3)
989     			ASSERT_EQUAL_(R.getRowCount(),3)
990     			{
991     				const Precision wx2 = (Precision)w[0]*w[0];
992     				const Precision wy2 = (Precision)w[1]*w[1];
993     				const Precision wz2 = (Precision)w[2]*w[2];
994     				R(0,0) = 1.0 - B*(wy2 + wz2);
995     				R(1,1) = 1.0 - B*(wx2 + wz2);
996     				R(2,2) = 1.0 - B*(wx2 + wy2);
997     			}
998     			{
999     				const Precision a = A*w[2];
1000     				const Precision b = B*(w[0]*w[1]);
1001     				R(0,1) = b - a;
1002     				R(1,0) = b + a;
1003     			}
1004     			{
1005     				const Precision a = A*w[1];
1006     				const Precision b = B*(w[0]*w[2]);
1007     				R(0,2) = b + a;
1008     				R(2,0) = b - a;
1009     			}
1010     			{
1011     				const Precision a = A*w[0];
1012     				const Precision b = B*(w[1]*w[2]);
1013     				R(1,2) = b - a;
1014     				R(2,1) = b + a;
1015     			}
1016     		}
