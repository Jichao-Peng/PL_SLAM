440     template<class TDescriptor, class F>
441     void TemplatedVocabulary<TDescriptor,F>::createScoringObject()
442     {
443       delete m_scoring_object;
444       m_scoring_object = NULL;
445       
446       switch(m_scoring)
447       {
448         case L1_NORM: 
449           m_scoring_object = new L1Scoring;
450           break;
451           
452         case L2_NORM:
453           m_scoring_object = new L2Scoring;
454           break;
455         
456         case CHI_SQUARE:
457           m_scoring_object = new ChiSquareScoring;
458           break;
459           
460         case KL:
461           m_scoring_object = new KLScoring;
462           break;
463           
464         case BHATTACHARYYA:
465           m_scoring_object = new BhattacharyyaScoring;
466           break;
467           
468         case DOT_PRODUCT:
469           m_scoring_object = new DotProductScoring;
470           break;
471         
472       }
473     }

535     template<class TDescriptor, class F>
536     void TemplatedVocabulary<TDescriptor,F>::create(
537       const std::vector<std::vector<TDescriptor> > &training_features)
538     {
539       m_nodes.clear();
540       m_words.clear();
541       
542       // expected_nodes = Sum_{i=0..L} ( k^i )
543     	int expected_nodes = 
544     		(int)((pow((double)m_k, (double)m_L + 1) - 1)/(m_k - 1));
545     
546       m_nodes.reserve(expected_nodes); // avoid allocations when creating the tree
547       
548       
549       std::vector<pDescriptor> features;
550       getFeatures(training_features, features);
551     
552     
553       // create root  
554       m_nodes.push_back(Node(0)); // root
555       
556       // create the tree
557       HKmeansStep(0, features, 1);
558     
559       // create the words
560       createWords();
561     
562       // and set the weight of each node of the tree
563       setNodeWeights(training_features);
564       
565     }

598     template<class TDescriptor, class F>
599     void TemplatedVocabulary<TDescriptor,F>::getFeatures(
600       const std::vector<std::vector<TDescriptor> > &training_features,
601       std::vector<pDescriptor> &features) const
602     {
603       features.resize(0);
604       
605       typename std::vector<std::vector<TDescriptor> >::const_iterator vvit;
606       typename std::vector<TDescriptor>::const_iterator vit;
607       for(vvit = training_features.begin(); vvit != training_features.end(); ++vvit)
608       {
609         features.reserve(features.size() + vvit->size());
610         for(vit = vvit->begin(); vit != vvit->end(); ++vit)
611         {
612           features.push_back(&(*vit));
613         }
614       }
615     }

619     template<class TDescriptor, class F>
620     void TemplatedVocabulary<TDescriptor,F>::HKmeansStep(NodeId parent_id, 
621       const std::vector<pDescriptor> &descriptors, int current_level)
622     {
623       if(descriptors.empty()) return;
624             
625       // features associated to each cluster
626       std::vector<TDescriptor> clusters;
627       std::vector<std::vector<unsigned int> > groups; // groups[i] = [j1, j2, ...]
628     	// j1, j2, ... indices of descriptors associated to cluster i
629     
630       clusters.reserve(m_k);
631     	groups.reserve(m_k);
632       
633       //const int msizes[] = { m_k, descriptors.size() };
634       //cv::SparseMat assoc(2, msizes, CV_8U);
635       //cv::SparseMat last_assoc(2, msizes, CV_8U);  
636       //// assoc.row(cluster_idx).col(descriptor_idx) = 1 iif associated
637       
638       if((int)descriptors.size() <= m_k)
639       {
640         // trivial case: one cluster per feature
641         groups.resize(descriptors.size());
642     
643         for(unsigned int i = 0; i < descriptors.size(); i++)
644         {
645           groups[i].push_back(i);
646           clusters.push_back(*descriptors[i]);
647         }
648       }
649       else
650       {
651         // select clusters and groups with kmeans
652         
653         bool first_time = true;
654         bool goon = true;
655         
656         // to check if clusters move after iterations
657         std::vector<int> last_association, current_association;
658     
659         while(goon)
660         {
661           // 1. Calculate clusters
662     
663     			if(first_time)
664     			{
665             // random sample 
666             initiateClusters(descriptors, clusters);
667           }
668           else
669           {
670             // calculate cluster centres
671     
672             for(unsigned int c = 0; c < clusters.size(); ++c)
673             {
674               std::vector<pDescriptor> cluster_descriptors;
675               cluster_descriptors.reserve(groups[c].size());
676               
677               /*
678               for(unsigned int d = 0; d < descriptors.size(); ++d)
679               {
680                 if( assoc.find<unsigned char>(c, d) )
681                 {
682                   cluster_descriptors.push_back(descriptors[d]);
683                 }
684               }
685               */
686               
687               std::vector<unsigned int>::const_iterator vit;
688               for(vit = groups[c].begin(); vit != groups[c].end(); ++vit)
689               {
690                 cluster_descriptors.push_back(descriptors[*vit]);
691               }
692               
693               
694               F::meanValue(cluster_descriptors, clusters[c]);
695             }
696             
697           } // if(!first_time)
698     
699           // 2. Associate features with clusters
700     
701           // calculate distances to cluster centers
702           groups.clear();
703           groups.resize(clusters.size(), std::vector<unsigned int>());
704           current_association.resize(descriptors.size());
705     
706           //assoc.clear();
707     
708           typename std::vector<pDescriptor>::const_iterator fit;
709           //unsigned int d = 0;
710           for(fit = descriptors.begin(); fit != descriptors.end(); ++fit)//, ++d)
711           {
712             double best_dist = F::distance(*(*fit), clusters[0]);
713             unsigned int icluster = 0;
714             
715             for(unsigned int c = 1; c < clusters.size(); ++c)
716             {
717               double dist = F::distance(*(*fit), clusters[c]);
718               if(dist < best_dist)
719               {
720                 best_dist = dist;
721                 icluster = c;
722               }
723             }
724     
725             //assoc.ref<unsigned char>(icluster, d) = 1;
726     
727             groups[icluster].push_back(fit - descriptors.begin());
728             current_association[ fit - descriptors.begin() ] = icluster;
729           }
730           
731           // kmeans++ ensures all the clusters has any feature associated with them
732     
733           // 3. check convergence
734           if(first_time)
735           {
736             first_time = false;
737           }
738           else
739           {
740             //goon = !eqUChar(last_assoc, assoc);
741             
742             goon = false;
743             for(unsigned int i = 0; i < current_association.size(); i++)
744             {
745               if(current_association[i] != last_association[i]){
746                 goon = true;
747                 break;
748               }
749             }
750           }
751     
752     			if(goon)
753     			{
754     				// copy last feature-cluster association
755     				last_association = current_association;
756     				//last_assoc = assoc.clone();
757     			}
758     			
759     		} // while(goon)
760         
761       } // if must run kmeans
762       
763       // create nodes
764       for(unsigned int i = 0; i < clusters.size(); ++i)
765       {
766         NodeId id = m_nodes.size();
767         m_nodes.push_back(Node(id));
768         m_nodes.back().descriptor = clusters[i];
769         m_nodes.back().parent = parent_id;
770         m_nodes[parent_id].children.push_back(id);
771       }
772       
773       // go on with the next level
774       if(current_level < m_L)
775       {
776         // iterate again with the resulting clusters
777         const std::vector<NodeId> &children_ids = m_nodes[parent_id].children;
778         for(unsigned int i = 0; i < clusters.size(); ++i)
779         {
780           NodeId id = children_ids[i];
781     
782           std::vector<pDescriptor> child_features;
783           child_features.reserve(groups[i].size());
784     
785           std::vector<unsigned int>::const_iterator vit;
786           for(vit = groups[i].begin(); vit != groups[i].end(); ++vit)
787           {
788             child_features.push_back(descriptors[*vit]);
789           }
790     
791           if(child_features.size() > 1)
792           {
793             HKmeansStep(id, child_features, current_level + 1);
794           }
795         }
796       }
797     }

811     template<class TDescriptor, class F>
812     void TemplatedVocabulary<TDescriptor,F>::initiateClustersKMpp(
813       const std::vector<pDescriptor> &pfeatures,
814         std::vector<TDescriptor> &clusters) const
815     {
816       // Implements kmeans++ seeding algorithm
817       // Algorithm:
818       // 1. Choose one center uniformly at random from among the data points.
819       // 2. For each data point x, compute D(x), the distance between x and the nearest 
820       //    center that has already been chosen.
821       // 3. Add one new data point as a center. Each point x is chosen with probability 
822       //    proportional to D(x)^2.
823       // 4. Repeat Steps 2 and 3 until k centers have been chosen.
824       // 5. Now that the initial centers have been chosen, proceed using standard k-means 
825       //    clustering.
826     
827       DUtils::Random::SeedRandOnce();
828     
829       clusters.resize(0);
830       clusters.reserve(m_k);
831       std::vector<double> min_dists(pfeatures.size(), std::numeric_limits<double>::max());
832       
833       // 1.
834       
835       int ifeature = DUtils::Random::RandomInt(0, pfeatures.size()-1);
836       
837       // create first cluster
838       clusters.push_back(*pfeatures[ifeature]);
839     
840       // compute the initial distances
841       typename std::vector<pDescriptor>::const_iterator fit;
842       std::vector<double>::iterator dit;
843       dit = min_dists.begin();
844       for(fit = pfeatures.begin(); fit != pfeatures.end(); ++fit, ++dit)
845       {
846         *dit = F::distance(*(*fit), clusters.back());
847       }  
848     
849       while((int)clusters.size() < m_k)
850       {
851         // 2.
852         dit = min_dists.begin();
853         for(fit = pfeatures.begin(); fit != pfeatures.end(); ++fit, ++dit)
854         {
855           if(*dit > 0)
856           {
857             double dist = F::distance(*(*fit), clusters.back());
858             if(dist < *dit) *dit = dist;
859           }
860         }
861         
862         // 3.
863         double dist_sum = std::accumulate(min_dists.begin(), min_dists.end(), 0.0);
864     
865         if(dist_sum > 0)
866         {
867           double cut_d;
868           do
869           {
870             cut_d = DUtils::Random::RandomValue<double>(0, dist_sum);
871           } while(cut_d == 0.0);
872     
873           double d_up_now = 0;
874           for(dit = min_dists.begin(); dit != min_dists.end(); ++dit)
875           {
876             d_up_now += *dit;
877             if(d_up_now >= cut_d) break;
878           }
879           
880           if(dit == min_dists.end()) 
881             ifeature = pfeatures.size()-1;
882           else
883             ifeature = dit - min_dists.begin();
884           
885           clusters.push_back(*pfeatures[ifeature]);
886     
887         } // if dist_sum > 0
888         else
889           break;
890           
891       } // while(used_clusters < m_k)
892     
893     }

898     void TemplatedVocabulary<TDescriptor,F>::createWords()
899     {
900       m_words.resize(0);
901       
902       if(!m_nodes.empty())
903       {
904         m_words.reserve( (int)pow((double)m_k, (double)m_L) );
905     
906         typename std::vector<Node>::iterator nit;
907         
908         nit = m_nodes.begin(); // ignore root
909         for(++nit; nit != m_nodes.end(); ++nit)
910         {
911           if(nit->isLeaf())
912           {
913             nit->word_id = m_words.size();
914             m_words.push_back( &(*nit) );
915           }
916         }
917       }
918     }

922     template<class TDescriptor, class F>
923     void TemplatedVocabulary<TDescriptor,F>::setNodeWeights
924       (const std::vector<std::vector<TDescriptor> > &training_features)
925     {
926       const unsigned int NWords = m_words.size();
927       const unsigned int NDocs = training_features.size();
928     
929       if(m_weighting == TF || m_weighting == BINARY)
930       {
931         // idf part must be 1 always
932         for(unsigned int i = 0; i < NWords; i++)
933           m_words[i]->weight = 1;
934       }
935       else if(m_weighting == IDF || m_weighting == TF_IDF)
936       {
937         // IDF and TF-IDF: we calculte the idf path now
938     
939         // Note: this actually calculates the idf part of the tf-idf score.
940         // The complete tf-idf score is calculated in ::transform
941     
942         std::vector<unsigned int> Ni(NWords, 0);
943         std::vector<bool> counted(NWords, false);
944         
945         typename std::vector<std::vector<TDescriptor> >::const_iterator mit;
946         typename std::vector<TDescriptor>::const_iterator fit;
947     
948         for(mit = training_features.begin(); mit != training_features.end(); ++mit)
949         {
950           fill(counted.begin(), counted.end(), false);
951     
952           for(fit = mit->begin(); fit < mit->end(); ++fit)
953           {
954             WordId word_id;
955             transform(*fit, word_id);
956     
957             if(!counted[word_id])
958             {
959               Ni[word_id]++;
960               counted[word_id] = true;
961             }
962           }
963         }
964     
965         // set ln(N/Ni)
966         for(unsigned int i = 0; i < NWords; i++)
967         {
968           if(Ni[i] > 0)
969           {
970             m_words[i]->weight = log((double)NDocs / (double)Ni[i]);
971           }// else // This cannot occur if using kmeans++
972         }
973       
974       }
975     
976     }

1045     template<class TDescriptor, class F>
1046     void TemplatedVocabulary<TDescriptor,F>::transform(
1047       const std::vector<TDescriptor>& features, BowVector &v) const
1048     {
1049       v.clear();
1050       
1051       if(empty())
1052       {
1053         return;
1054       }
1055     
1056       // normalize 
1057       LNorm norm;
1058       bool must = m_scoring_object->mustNormalize(norm);
1059     
1060       typename std::vector<TDescriptor>::const_iterator fit;
1061     
1062       if(m_weighting == TF || m_weighting == TF_IDF)
1063       {
1064         for(fit = features.begin(); fit < features.end(); ++fit)
1065         {
1066           WordId id;
1067           WordValue w; 
1068           // w is the idf value if TF_IDF, 1 if TF
1069           
1070           transform(*fit, id, w);
1071           
1072           // not stopped
1073           if(w > 0) v.addWeight(id, w);
1074         }
1075         
1076         if(!v.empty() && !must)
1077         {
1078           // unnecessary when normalizing
1079           const double nd = v.size();
1080           for(BowVector::iterator vit = v.begin(); vit != v.end(); vit++) 
1081             vit->second /= nd;
1082         }
1083         
1084       }
1085       else // IDF || BINARY
1086       {
1087         for(fit = features.begin(); fit < features.end(); ++fit)
1088         {
1089           WordId id;
1090           WordValue w;
1091           // w is idf if IDF, or 1 if BINARY
1092           
1093           transform(*fit, id, w);
1094           
1095           // not stopped
1096           if(w > 0) v.addIfNotExist(id, w);
1097           
1098         } // if add_features
1099       } // if m_weighting == ...
1100       
1101       if(must) v.normalize(norm);
1102     }

1106     template<class TDescriptor, class F> 
1107     void TemplatedVocabulary<TDescriptor,F>::transform(
1108       const std::vector<TDescriptor>& features,
1109       BowVector &v, FeatureVector &fv, int levelsup) const
1110     {
1111       v.clear();
1112       fv.clear();
1113       
1114       if(empty()) // safe for subclasses
1115       {
1116         return;
1117       }
1118       
1119       // normalize 
1120       LNorm norm;
1121       bool must = m_scoring_object->mustNormalize(norm);
1122       
1123       typename std::vector<TDescriptor>::const_iterator fit;
1124       
1125       if(m_weighting == TF || m_weighting == TF_IDF)
1126       {
1127         unsigned int i_feature = 0;
1128         for(fit = features.begin(); fit < features.end(); ++fit, ++i_feature)
1129         {
1130           WordId id;
1131           NodeId nid;
1132           WordValue w; 
1133           // w is the idf value if TF_IDF, 1 if TF
1134           
1135           transform(*fit, id, w, &nid, levelsup);
1136           
1137           if(w > 0) // not stopped
1138           { 
1139             v.addWeight(id, w);
1140             fv.addFeature(nid, i_feature);
1141           }
1142         }
1143         
1144         if(!v.empty() && !must)
1145         {
1146           // unnecessary when normalizing
1147           const double nd = v.size();
1148           for(BowVector::iterator vit = v.begin(); vit != v.end(); vit++) 
1149             vit->second /= nd;
1150         }
1151       
1152       }
1153       else // IDF || BINARY
1154       {
1155         unsigned int i_feature = 0;
1156         for(fit = features.begin(); fit < features.end(); ++fit, ++i_feature)
1157         {
1158           WordId id;
1159           NodeId nid;
1160           WordValue w;
1161           // w is idf if IDF, or 1 if BINARY
1162           
1163           transform(*fit, id, w, &nid, levelsup);
1164           
1165           if(w > 0) // not stopped
1166           {
1167             v.addIfNotExist(id, w);
1168             fv.addFeature(nid, i_feature);
1169           }
1170         }
1171       } // if m_weighting == ...
1172       
1173       if(must) v.normalize(norm);
1174     }

1197     template<class TDescriptor, class F>
1198     void TemplatedVocabulary<TDescriptor,F>::transform(const TDescriptor &feature, 
1199       WordId &word_id, WordValue &weight, NodeId *nid, int levelsup) const
1200     { 
1201       // propagate the feature down the tree
1202       std::vector<NodeId> nodes;
1203       typename std::vector<NodeId>::const_iterator nit;
1204     
1205       // level at which the node must be stored in nid, if given
1206       const int nid_level = m_L - levelsup;
1207       if(nid_level <= 0 && nid != NULL) *nid = 0; // root
1208     
1209       NodeId final_id = 0; // root
1210       int current_level = 0;
1211     
1212       do
1213       {
1214         ++current_level;
1215         nodes = m_nodes[final_id].children;
1216         final_id = nodes[0];
1217      
1218         double best_d = F::distance(feature, m_nodes[final_id].descriptor);
1219     
1220         for(nit = nodes.begin() + 1; nit != nodes.end(); ++nit)
1221         {
1222           NodeId id = *nit;
1223           double d = F::distance(feature, m_nodes[id].descriptor);
1224           if(d < best_d)
1225           {
1226             best_d = d;
1227             final_id = id;
1228           }
1229         }
1230         
1231         if(nid != NULL && current_level == nid_level)
1232           *nid = final_id;
1233         
1234       } while( !m_nodes[final_id].isLeaf() );
1235     
1236       // turn node id into word id
1237       word_id = m_nodes[final_id].word_id;
1238       weight = m_nodes[final_id].weight;
1239     }

1258     template<class TDescriptor, class F>
1259     void TemplatedVocabulary<TDescriptor,F>::getWordsFromNode
1260       (NodeId nid, std::vector<WordId> &words) const
1261     {
1262       words.clear();
1263       
1264       if(m_nodes[nid].isLeaf())
1265       {
1266         words.push_back(m_nodes[nid].word_id);
1267       }
1268       else
1269       {
1270         words.reserve(m_k); // ^1, ^2, ...
1271         
1272         std::vector<NodeId> parents;
1273         parents.push_back(nid);
1274         
1275         while(!parents.empty())
1276         {
1277           NodeId parentid = parents.back();
1278           parents.pop_back();
1279           
1280           const std::vector<NodeId> &child_ids = m_nodes[parentid].children;
1281           std::vector<NodeId>::const_iterator cit;
1282           
1283           for(cit = child_ids.begin(); cit != child_ids.end(); ++cit)
1284           {
1285             const Node &child_node = m_nodes[*cit];
1286             
1287             if(child_node.isLeaf())
1288               words.push_back(child_node.word_id);
1289             else
1290               parents.push_back(*cit);
1291             
1292           } // for each child
1293         } // while !parents.empty
1294       }
1295     }

1414     
1415       // words
1416       f << "words" << "[";
1417       
1418       typename std::vector<Node*>::const_iterator wit;
1419       for(wit = m_words.begin(); wit != m_words.end(); wit++)
1420       {
1421         WordId id = wit - m_words.begin();
1422         f << "{:";
1423         f << "wordId" << (int)id;
1424         f << "nodeId" << (int)(*wit)->id;

1427       
1428       f << "]"; // words
1429     
1430       f << "}";
1431     
1432     }
