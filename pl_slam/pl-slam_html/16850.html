36     SlamConfig::SlamConfig()
37         : Config()
38     {
39         // Note: do not set default values for StVO Config here! (they can be overriten by the default class constructor...)
40     
41         // SLAM parameters
42         // -----------------------------------------------------------------------------------------------------
43         fast_matching         = false;      // allow for the fast matching (window-based) of the map features
44         has_refinement        = false;      // refine the pose between keyframes (disabled as it is also performed by the LBA)
45         mutithread_slam       = true;       // if true the system runs with both the VO, LBA and LC in parallel threads
46     
47         // lm numbers and errors
48         min_lm_obs            = 5;          // min number of observations for a landmark to be considered as inlier
49         max_common_fts_kf     = 0.9;        // max number of common features for a keyframe to be considered redundant (disabled)
50     
51         max_kf_epip_p         = 1.0;        // max epip distance for points in LBA
52         max_kf_epip_l         = 1.0;        // max epip distance for line segments in LBA
53     
54         max_point_point_error = 0.1;        // maximum distance to fuse landmarks from local map (points)
55         max_point_line_error  = 0.1;        // maximum distance to fuse landmarks from local map (lines)
56         max_dir_line_error    = 0.1;        // maximum angle in line direction to fuse landmarks from local map
57         kf_inlier_ratio       = 30.0;       // ratio of inliers to fuse local landmarks
58     
59         // graphs parameters
60         min_lm_ess_graph      = 150;        // minimum number of landmarks for connectivity in Essential graph
61         min_lm_cov_graph      = 75;         // minimum number of landmarks for connectivity in Covisibility graph
62         min_kf_local_map      = 3;          // min number of landmarks for the local mapping
63     
64         // LBA
65         lambda_lba_lm         = 0.00001;    // (if auto, this is the initial tau)
66         lambda_lba_k          = 10.0;       // lambda_k for LM method in LBA
67         max_iters_lba         = 15;         // maximum number of iterations
68     
69         // Loop closure
70         vocabulary_p          = "/media/zhijian/Document/grow/slam/PL_SLAM/pl_slam/vocabulary/mapir_orb.yml";
71         vocabulary_l          = "/media/zhijian/Document/grow/slam/PL_SLAM/pl_slam/vocabulary/mapir_lsd.yml";
72     
73         lc_res                = 1.0;        // maximum residue in relative pose estimation
74         lc_unc                = 0.01;       // maximum uncertainty value in relative pose estimation
75         lc_inl                = 0.3;        // minimum inlier ratio in relative pose estimation
76         lc_trs                = 1.5;        // maximum translation in relative pose estimation
77         lc_rot                = 35.0;       // maximum rotation in relative pose estimation
78     
79         max_iters_pgo         = 100;        // maximum number of iterations of the PGO
80         lc_kf_dist            = 50;         // minimum number of KFs from prev LC
81         lc_kf_max_dist        = 50;         // max distance from last LC KF
82         lc_nkf_closest        = 4;          // number of KFs closest to the match to consider it as positive
83         lc_inlier_ratio       = 30.0;       // inlier ratio to consider or not a loop closure
84     
85         min_pt_matches        = 10;         // min number of point observations
86         min_ls_matches        = 6;          // min number of line segment observations
87     }

97     template<typename T>
98     inline T loadSafe(const YAML::Node &config, std::string param, T default_value = T()) {
99     
100         if (YAML::Node parameter = config[param])
101             return parameter.as<T>();
102         else
103             return default_value;
104     }

106     void SlamConfig::loadFromFile( const string &config_file )
107     {
108         if (!boost::filesystem::exists(config_file) || !boost::filesystem::is_regular(config_file)) {
109             cout << "[SlamConfig->loadFromFile] Invalid config file, keeping default params..." << endl;
110             return;
111         }
112     
113         // StVO-PL options
114         // -----------------------------------------------------------------------------------------------------
115         Config::loadFromFile(config_file);
116     
117         // PL-SLAM options
118         // -----------------------------------------------------------------------------------------------------
119         YAML::Node config = YAML::LoadFile(config_file);
120     
121         SlamConfig::minLMObs() = loadSafe(config, "min_lm_obs", SlamConfig::minLMObs());
122         SlamConfig::maxCommonFtsKF() = loadSafe(config, "max_common_fts_kf", SlamConfig::maxCommonFtsKF());
123     
124         SlamConfig::maxKFEpipP() = loadSafe(config, "max_kf_epip_p", SlamConfig::maxKFEpipP());
125         SlamConfig::maxKFEpipL() = loadSafe(config, "max_kf_epip_l", SlamConfig::maxKFEpipL());
126     
127         SlamConfig::maxLM3DErr() = loadSafe(config, "max_lm_3d_err", SlamConfig::maxLM3DErr());
128         SlamConfig::maxLMDirErr() = loadSafe(config, "max_lm_dir_err", SlamConfig::maxLMDirErr());
129         SlamConfig::maxPointPointError() = loadSafe(config, "max_point_point_error", SlamConfig::maxPointPointError());
130         SlamConfig::maxPointLineError() = loadSafe(config, "max_point_line_error", SlamConfig::maxPointLineError());
131         SlamConfig::maxDirLineError() = loadSafe(config, "max_dir_line_error", SlamConfig::maxDirLineError());
132     
133         SlamConfig::minLMEssGraph() = loadSafe(config, "min_lm_ess_graph", SlamConfig::minLMEssGraph());
134         SlamConfig::minLMCovGraph() = loadSafe(config, "min_lm_cov_graph", SlamConfig::minLMCovGraph());
135         SlamConfig::minKFLocalMap() = loadSafe(config, "min_kf_local_map", SlamConfig::minKFLocalMap());
136     
137         SlamConfig::lambdaLbaLM() = loadSafe(config, "lambda_lba_lm", SlamConfig::lambdaLbaLM());
138         SlamConfig::lambdaLbaK() = loadSafe(config, "lambda_lba_k", SlamConfig::lambdaLbaK());
139         SlamConfig::maxItersLba() = loadSafe(config, "max_iters_lba", SlamConfig::maxItersLba());
140     
141         SlamConfig::dbowVocP() = loadSafe(config, "vocabulary_p", SlamConfig::dbowVocP());
142         SlamConfig::dbowVocL() = loadSafe(config, "vocabulary_l", SlamConfig::dbowVocL());
143     
144         SlamConfig::lcMat() = loadSafe(config, "lc_mat", SlamConfig::lcMat());
145         SlamConfig::lcRes() = loadSafe(config, "lc_res", SlamConfig::lcRes());
146         SlamConfig::lcUnc() = loadSafe(config, "lc_unc", SlamConfig::lcUnc());
147         SlamConfig::lcInl() = loadSafe(config, "lc_inl", SlamConfig::lcInl());
148         SlamConfig::lcTrs() = loadSafe(config, "lc_trs", SlamConfig::lcTrs());
149         SlamConfig::lcRot() = loadSafe(config, "lc_rot", SlamConfig::lcRot());
150     
151         SlamConfig::maxItersPGO() = loadSafe(config, "max_iters_pgo", SlamConfig::maxItersPGO());
152         SlamConfig::lcKFDist() = loadSafe(config, "lc_kf_dist", SlamConfig::lcKFDist());
153         SlamConfig::lcKFMaxDist() = loadSafe(config, "lc_kf_max_dist", SlamConfig::lcKFMaxDist());
154         SlamConfig::lcNKFClosest() = loadSafe(config, "lc_nkf_closest", SlamConfig::lcNKFClosest());
155         SlamConfig::lcInlierRatio() = loadSafe(config, "lc_inlier_ratio", SlamConfig::lcInlierRatio());
156     
157         SlamConfig::minPointMatches() = loadSafe(config, "min_pt_matches", SlamConfig::minPointMatches());
158         SlamConfig::minLineMatches() = loadSafe(config, "min_ls_matches", SlamConfig::minLineMatches());
159         SlamConfig::kfInlierRatio() = loadSafe(config, "kf_inlier_ratio", SlamConfig::kfInlierRatio());
160     
161         SlamConfig::fastMatching() = loadSafe(config, "fast_matching", SlamConfig::fastMatching());
162         SlamConfig::hasRefinement() = loadSafe(config, "has_refinement", SlamConfig::hasRefinement());
163         SlamConfig::multithreadSLAM() = loadSafe(config, "mutithread_slam", SlamConfig::multithreadSLAM());
164     }
