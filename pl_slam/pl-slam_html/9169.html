31     KeyFrame::KeyFrame( const StereoFrame* sf )
32     {
33         int i=0;
34         kf_idx    = -1;
35         x_kf_w    = logmap_se3( T_kf_w );
36     
37         T_kf_w    = sf->Tfw ;
38         xcov_kf_w = sf->Tfw_cov;
39     
40         //这里并没有直接使用指针赋值，而是把内容复制了一遍
41         stereo_frame = new StereoFrame( sf->img_l, sf->img_r, kf_idx, sf->cam );
42         stereo_frame->pdesc_l   = sf->pdesc_l;
43         stereo_frame->pdesc_r   = sf->pdesc_r;
44         stereo_frame->ldesc_l   = sf->ldesc_l;
45         stereo_frame->ldesc_r   = sf->ldesc_r;
46     
47         std::vector<PointFeature*> &stereo_pt = stereo_frame->stereo_pt;
48         stereo_pt.resize(sf->stereo_pt.size());
49         for (int idx = 0; idx < stereo_pt.size(); ++idx)
50             stereo_pt[idx] = sf->stereo_pt[idx]->safeCopy();
51     
52         std::vector<LineFeature*> &stereo_ls = stereo_frame->stereo_ls;
53         stereo_ls.resize(sf->stereo_ls.size());
54         for (int idx = 0; idx < stereo_ls.size(); ++idx)
55             stereo_ls[idx] = sf->stereo_ls[idx]->safeCopy();
56     }

88     Mat KeyFrame::plotKeyFrame()
89     {
90         // create new image to modify it
91         Mat img_l_aux;
92         stereo_frame->img_l.copyTo( img_l_aux );
93         if( img_l_aux.channels() == 1 )
94             cvtColor(img_l_aux, img_l_aux, CV_GRAY2BGR, 3);
95         else if (img_l_aux.channels() == 4)
96             cvtColor(img_l_aux, img_l_aux, CV_BGRA2BGR, 3);
97         else if (img_l_aux.channels() != 3)
98             throw std::runtime_error(std::string("[KeyFrame->plotKeyFrame] unsupported image format: ") +
99                                      std::to_string(img_l_aux.channels()));
100         img_l_aux.convertTo(img_l_aux, CV_8UC3);
101     
102         // Variables
103         unsigned int    r = 0, g, b = 0;
104         Point2f         p,q;
105         double          thick = 1.5;
106         int             k = 0, radius  = 3;
107         // plot point features
108         for( vector<PointFeature*>::iterator pt_it = stereo_frame->stereo_pt.begin(); pt_it != stereo_frame->stereo_pt.end(); pt_it++)
109         {
110             if( (*pt_it)->idx != -1 )
111             {
112                 g = 200;
113                 p = cv::Point( int((*pt_it)->pl(0)), int((*pt_it)->pl(1)) );
114                 circle( img_l_aux, p, radius, Scalar(b,g,r), thick);
115             }
116         }
117         // plot line segment features
118         for( vector<LineFeature*>::iterator ls_it = stereo_frame->stereo_ls.begin(); ls_it != stereo_frame->stereo_ls.end(); ls_it++)
119         {
120             if( (*ls_it)->idx != -1 )
121             {
122                 g = 200;
123                 p = cv::Point( int((*ls_it)->spl(0)), int((*ls_it)->spl(1)) );
124                 q = cv::Point( int((*ls_it)->epl(0)), int((*ls_it)->epl(1)) );
125                 line( img_l_aux, p, q, Scalar(b,g,r), thick);
126             }
127         }
128     
129         return img_l_aux;
130     }
