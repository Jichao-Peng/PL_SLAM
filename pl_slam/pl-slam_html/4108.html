42     
43     //#define NO_SECENE
44     
45     int main(int argc, char **argv)
46     {
47     
48         // read inputs
49         string dataset_name, config_file;
50         int frame_offset = 0, frame_number = 0, frame_step = 1;
51         /*
52          dataset_name 数据集名字 这里为zero
53          frame_offset 跳过前多少个图像帧
54          frame_number 只考虑这么多对帧
55          frame_step   每多少个帧处理一次
56          config_file 配置文件,默认配置为slamconfig
57          */
58         if (!getInputArgs(argc, argv, dataset_name, frame_offset, frame_number, frame_step, config_file)) {
59             //参数检测错误时报错
60             showHelp();
61             return -1;
62         }
63         
64         //显示传入参数
65         cout << "" <<endl;
66         for(int i=0;i<argc;i++)
67             cout << i <<"  "<<argv[i] <<endl;
68         
69         //传入配置文件的数据
70         if (!config_file.empty()) 
71         {
72             SlamConfig::loadFromFile(config_file);
73             cout<<config_file<<endl;
74         }
75     
76         //传入点词典
77         if (SlamConfig::hasPoints() &&
78                 (!boost::filesystem::exists(SlamConfig::dbowVocP()) || !boost::filesystem::is_regular_file(SlamConfig::dbowVocP()))) {
79             cout << "Invalid vocabulary for points" << endl;
80             return -1;
81         }
82     
83         //传入线词典
84         if (SlamConfig::hasLines() &&
85                 (!boost::filesystem::exists(SlamConfig::dbowVocL()) || !boost::filesystem::is_regular_file(SlamConfig::dbowVocL()))) {
86             cout << "Invalid vocabulary for lines" << endl;
87             return -1;
88         }
89     
90         // read dataset root dir fron environment variable
91         boost::filesystem::path dataset_path(string( "/media/zhijian/Document/grow/slam/slamDataSet/KITTI/color"));
92         
93         if (!boost::filesystem::exists(dataset_path) || !boost::filesystem::is_directory(dataset_path)) {
94             cout << "Check your DATASETS_DIR environment variable" << endl;
95             return -1;
96         }
97     
98         //添加具体数据集
99         dataset_path /= dataset_name;
100         if (!boost::filesystem::exists(dataset_path) || !boost::filesystem::is_directory(dataset_path)) {
101             cout << "Invalid dataset path" << endl;
102             return -1;
103         }
104         cout << endl << "Initializing PL-SLAM...." << flush;
105     
106         string dataset_dir = dataset_path.string();
107         //传入相机配置文件，建立相机模型
108         PinholeStereoCamera*  cam_pin = new PinholeStereoCamera((dataset_path / "dataset_params.yaml").string());
109         //数据集类
110         Dataset dataset(dataset_dir, *cam_pin, frame_offset, frame_number, frame_step);
111     
112         // create scene 
113         string scene_cfg_name;
114         if( (dataset_name.find("kitti")!=std::string::npos) ||
115             (dataset_name.find("malaga")!=std::string::npos)  )
116         {
117             scene_cfg_name = "../config/scene_config.ini";
118         }
119         else{
120             //执行这一步
121             scene_cfg_name = "../config/scene_config_indoor.ini";
122         }
123         //场景类
124         #ifndef NO_SECENE
125             slamScene scene(scene_cfg_name);
126         #endif
127         Matrix4d Tcw, Tfw = Matrix4d::Identity();
128         Tcw = Matrix4d::Identity();
129         
130         #ifndef NO_SECENE
131             scene.setStereoCalibration( cam_pin->getK(), cam_pin->getB() );
132             scene.initializeScene(Tfw);
133         #endif
134     
135         // create PLSLAM object
136         PLSLAM::MapHandler* map = new PLSLAM::MapHandler(cam_pin);
137     
138         cout << " ... done. " << endl;
139     
140         Timer timer;
141     
142         // initialize and run PL-StVO
143         int frame_counter = 0;
144         StereoFrameHandler* StVO = new StereoFrameHandler(cam_pin);
145         Mat img_l, img_r;
146         ofstream fout("/media/zhijian/Document/grow/slam/slamDataSet/KITTI/data_odometry_poses/dataset/poses/my00.txt");
147         //当数据集不空时
148         while (dataset.nextFrame(img_l, img_r))
149         {
150             if( frame_counter == 0 ) // initialize
151             {
152                 //双目的初始化
153                 StVO->initialize(img_l,img_r,0);
154                 //第一帧肯定是关键帧
155                 PLSLAM::KeyFrame* kf = new PLSLAM::KeyFrame( StVO->prev_frame, 0 );
156                 //地图初始化
157                 map->initialize( kf );
158                 // update scene
159                 #ifndef NO_SECENE
160                     scene.initViewports( img_l.cols, img_r.rows );
161                     //更新raw image
162                     scene.setImage(StVO->prev_frame->plotStereoFrame());
163                     scene.updateSceneSafe( map );
164                 #endif
165             }
166             else // run
167             {
168                 // PL-StVO
169                 timer.start();
170                 //frame_counter>0
171                 StVO->insertStereoPair( img_l, img_r, frame_counter );
172                 StVO->optimizePose();
173                 double t1 = timer.stop(); //ms
174     //             cout << "------------------------------------------   Frame #" << frame_counter
175     //                  << "   ----------------------------------------" << endl;
176     //             cout << endl << "VO Runtime: " << t1 << endl;
177     
178                 // check if a new keyframe is 
179                 
180     
181                 if( StVO->needNewKF() )
182                 {
183     //                 cout <<         "#KeyFrame:     " << map->max_kf_idx + 1;
184     //                 cout << endl << "#Points:       " << map->map_points.size();
185     //                 cout << endl << "#Segments:     " << map->map_lines.size();
186     //                 cout << endl << endl;
187     
188                     // grab StF and update KF in StVO (the StVO thread can continue after this point)
189                     PLSLAM::KeyFrame* curr_kf = new PLSLAM::KeyFrame( StVO->curr_frame );
190                     // update KF in StVO
191                     StVO->currFrameIsKF();
192                     map->addKeyFrame( curr_kf );
193                     // update scene
194                     #ifndef NO_SECENE
195                         // plotStereoFrame() 把点线特征给画上去
196                         scene.setImage(StVO->curr_frame->plotStereoFrame());
197                         scene.updateSceneSafe( map );
198                     #endif
199                     Matrix3d R = curr_kf->T_kf_w.block(0,0,3,3).transpose();
200                     vector<float> q = toQuaternion(R);
201                     fout << setprecision(7) << " "<< curr_kf->T_kf_w(0, 3) << " " << curr_kf->T_kf_w(1, 3) << " " << curr_kf->T_kf_w(2, 3) << " "<< q[0] << " " << q[1] << " " << q[2] << " " << q[3] << endl;    
202                 }
203                 else
204                 {
205                     #ifndef NO_SECENE
206                         scene.setImage(StVO->curr_frame->plotStereoFrame());
207                         scene.setPose( StVO->curr_frame->DT );
208                         scene.updateScene();
209                     #endif
210                     Matrix3d R = StVO->curr_frame->Tfw.block(0,0,3,3).transpose();
211                     vector<float> q = toQuaternion(R);
212                     fout << setprecision(7) << " "<< StVO->curr_frame->Tfw(0, 3) << " " << StVO->curr_frame->Tfw(1, 3) << " " << StVO->curr_frame->Tfw(2, 3) << " "<< q[0] << " " << q[1] << " " << q[2] << " " << q[3] << endl;
213                 }
214             
215                 // update StVO
216             StVO->updateFrame();
217             }
218     
219             frame_counter++;
220         }
221     
222         fout.close();
223         // finish SLAM
224         map->finishSLAM();
225         
226         #ifndef NO_SECENE
227             scene.updateScene( map );
228         #endif
229         // perform GBA
230         cout << endl << "Performing Global Bundle Adjustment..." ;
231         map->globalBundleAdjustment();
232         cout << " ... done." << endl;
233         #ifndef NO_SECENE
234             scene.updateSceneGraphs( map );
235         #endif
236     
237         // wait until the scene is closed
238         #ifndef NO_SECENE
239             while( scene.isOpen() );
240         #endif
241     
242         return 0;
243     }

245     void showHelp() {
246         cout << endl << "Usage: ./imgPLSLAM <dataset_name> [options]" << endl
247              << "Options:" << endl
248              << "\t-o Offset (number of frames to skip in the dataset directory" << endl
249              << "\t-n Number of frames to process the sequence" << endl
250              << "\t-s Parameter to skip s-1 frames (default 1)" << endl
251              << "\t-c Config file" << endl
252              << endl;
253     }
254     
255     bool getInputArgs(int argc, char **argv, std::string &dataset_name, int &frame_offset, int &frame_number, int &frame_step, std::string &config_file) {
256     
257         if( argc < 2 || argc > 10 || (argc % 2) == 1 )
258             return false;
259     
260         dataset_name = argv[1];
261         int nargs = argc/2 - 1;
262         for( int i = 0; i < nargs; i++ )
263         {
264             int j = 2*i + 2;
265             if( string(argv[j]) == "-o" )
266                 frame_offset = stoi(argv[j+1]);
267             else if( string(argv[j]) == "-n" )
268                 frame_number = stoi(argv[j+1]);
269             else if( string(argv[j]) == "-s" )
270                 frame_step = stoi(argv[j+1]);
271             else if (string(argv[j]) == "-c")
272                 config_file = string(argv[j+1]);
273             else
274                 return false;
275         }
276     
277         return true;
278     }
