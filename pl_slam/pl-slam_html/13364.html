25     	/** An implementation of the Levenberg-Marquardt algorithm for least-square minimization.
26     	 *
27     	 *  Refer to this <a href="http://www.mrpt.org/Levenberg%E2%80%93Marquardt_algorithm" >page</a> for more details on the algorithm and its usage.
28     	 *
29     	 * \tparam NUMTYPE The numeric type for all the operations (float, double, or long double)
30     	 * \tparam USERPARAM The type of the "y" input to the user supplied evaluation functor. Default type is a vector of NUMTYPE.
31     	 * \ingroup mrpt_base_grp
32     	 */
33     	template <typename VECTORTYPE = Eigen::VectorXd, class USERPARAM = VECTORTYPE >
34     	class CLevenbergMarquardtTempl : public mrpt::utils::CDebugOutputCapable
35     	{
36     	public:
37     		typedef typename VECTORTYPE::Scalar  NUMTYPE;
38     		typedef Eigen::Matrix<NUMTYPE,Eigen::Dynamic,Eigen::Dynamic>  matrix_t;
39     		typedef VECTORTYPE vector_t;
40     
41     
42     		/** The type of the function passed to execute. The user must supply a function which evaluates the error of a given point in the solution space.
43     		  *  \param x The state point under examination.
44     		  *  \param y The same object passed to "execute" as the parameter "userParam".
45     		  *  \param out The vector of (non-squared) errors, of the average square root error, for the given "x". The functor code must set the size of this vector.
46     		  */
47     		typedef void (*TFunctorEval)(
48     			const VECTORTYPE &x,
49     			const USERPARAM &y,
50     			VECTORTYPE &out);
51     
52     		/** The type of an optional functor passed to \a execute to replace the Euclidean addition "x_new = x_old + x_incr" by any other operation.
53     		  */
54     		typedef void (*TFunctorIncrement)(
55     			VECTORTYPE &x_new,
56     			const VECTORTYPE &x_old,
57     			const VECTORTYPE &x_incr,
58     			const USERPARAM &user_param);
59     
60     		struct TResultInfo
61     		{
62     			NUMTYPE		final_sqr_err;
63     			size_t		iterations_executed;
64     			VECTORTYPE	last_err_vector;		//!< The last error vector returned by the user-provided functor.
65     			matrix_t	path;	//!< Each row is the optimized value at each iteration.
66     
67     			/** This matrix can be used to obtain an estimate of the optimal parameters covariance matrix:
68     			  *  \f[ COV = H M H^\top \f]
69     			  *  With COV the covariance matrix of the optimal parameters, H this matrix, and M the covariance of the input (observations).
70     			  */
71      			matrix_t H;
72     		};
73     
74     		/** Executes the LM-method, with derivatives estimated from
75     		  *  \a functor is a user-provided function which takes as input two vectors, in this order:
76     		  *		- x: The parameters to be optimized.
77     		  *		- userParam: The vector passed to the LM algorithm, unmodified.
78     		  *	  and must return the "error vector", or the error (not squared) in each measured dimension, so the sum of the square of that output is the overall square error.
79     		  *
80     		  * \a x_increment_adder Is an optional functor which may replace the Euclidean "x_new = x + x_increment" at the core of the incremental optimizer by
81     		  *     any other operation. It can be used for example, in on-manifold optimizations.
82     		  */
83     		static void	execute(
84     			VECTORTYPE			&out_optimal_x,
85     			const VECTORTYPE	&x0,
86     			TFunctorEval		functor,
87     			const VECTORTYPE	&increments,
88     			const USERPARAM		&userParam,
89     			TResultInfo			&out_info,
90     			bool				verbose = false,
91     			const size_t		maxIter = 200,
92     			const NUMTYPE		tau = 1e-3,
93     			const NUMTYPE		e1 = 1e-8,
94     			const NUMTYPE		e2 = 1e-8,
95     			bool                returnPath =true,
96     			TFunctorIncrement	x_increment_adder = NULL
97     			)
98     		{
99     			using namespace mrpt;
100     			using namespace mrpt::utils;
101     			using namespace mrpt::math;
102     			using namespace std;
103     
104     			MRPT_START
105     
106     			VECTORTYPE &x=out_optimal_x; // Var rename
107     
108     			// Asserts:
109     			ASSERT_( increments.size() == x0.size() );
110     
111     			x=x0;									// Start with the starting point
112     			VECTORTYPE	f_x;					// The vector error from the user function
113     			matrix_t AUX;
114     			matrix_t J;		// The Jacobian of "f"
115     			VECTORTYPE	g;						// The gradient
116     
117     			// Compute the jacobian and the Hessian:
118     			mrpt::math::estimateJacobian( x, functor, increments, userParam, J);
119     			out_info.H.multiply_AtA(J);
120     
121     			const size_t  H_len = out_info.H.getColCount();
122     
123     			// Compute the gradient:
124     			functor(x, userParam ,f_x);
125     			J.multiply_Atb(f_x, g);
126     
127     			// Start iterations:
128     			bool	found = math::norm_inf(g)<=e1;
129     			if (verbose && found)	printf_debug("[LM] End condition: math::norm_inf(g)<=e1 :%f\n",math::norm_inf(g));
130     
131     			NUMTYPE	lambda = tau * out_info.H.maximumDiagonal();
132     			size_t  iter = 0;
133     			NUMTYPE	v = 2;
134     
135     			VECTORTYPE	h_lm;
136     			VECTORTYPE	xnew, f_xnew ;
137     			NUMTYPE			F_x  = pow( math::norm( f_x ), 2);
138     
139     			const size_t	N = x.size();
140     
141     			if (returnPath)	{
142     				out_info.path.setSize(maxIter,N+1);
143     				out_info.path.block(iter,0,1,N) = x.transpose();
144     			}	else out_info.path = Eigen::Matrix<NUMTYPE,Eigen::Dynamic,Eigen::Dynamic>(); // Empty matrix
145     
146     			while (!found && ++iter<maxIter)
147     			{
148     				// H_lm = -( H + \lambda I ) ^-1 * g
149     				matrix_t H = out_info.H;
150     				for (size_t k=0;k<H_len;k++)
151     					H(k,k)+= lambda;
152     
153     				H.inv_fast(AUX);
154     				AUX.multiply_Ab(g,h_lm);
155     				h_lm *= NUMTYPE(-1.0);
156     
157     				double h_lm_n2 = math::norm(h_lm);
158     				double x_n2 = math::norm(x);
159     
160     				if (verbose) printf_debug("[LM] Iter:%u x=%s\n",(unsigned)iter,sprintf_vector(" %f",x).c_str() );
161     
162     				if (h_lm_n2<e2*(x_n2+e2))
163     				{
164     					// Done:
165     					found = true;
166     					if (verbose) printf_debug("[LM] End condition: %e < %e\n", h_lm_n2, e2*(x_n2+e2) );
167     				}
168     				else
169     				{
170     					// Improvement: xnew = x + h_lm;
171     					if (!x_increment_adder)
172     							xnew = x + h_lm; // Normal Euclidean space addition.
173     					else 	x_increment_adder(xnew, x, h_lm, userParam);
174     
175     					functor(xnew, userParam ,f_xnew );
176     					const double F_xnew = pow( math::norm(f_xnew), 2);
177     
178     					// denom = h_lm^t * ( \lambda * h_lm - g )
179     					VECTORTYPE	tmp(h_lm);
180     					tmp *= lambda;
181     					tmp -= g;
182     					tmp.array() *=h_lm.array();
183     					double denom = tmp.sum();
184     					double l = (F_x - F_xnew) / denom;
185     
186     					if (l>0) // There is an improvement:
187     					{
188     						// Accept new point:
189     						x   = xnew;
190     						f_x = f_xnew;
191     						F_x = F_xnew;
192     
193     						math::estimateJacobian( x, functor, increments, userParam, J);
194     						out_info.H.multiply_AtA(J);
195     						J.multiply_Atb(f_x, g);
196     
197     						found = math::norm_inf(g)<=e1;
198     						if (verbose && found)	printf_debug("[LM] End condition: math::norm_inf(g)<=e1 : %e\n", math::norm_inf(g) );
199     
200     						lambda *= max(0.33, 1-pow(2*l-1,3) );
201     						v = 2;
202     					}
203     					else
204     					{
205     						// Nope...
206     						lambda *= v;
207     						v*= 2;
208     					}
209     
210     
211     					if (returnPath)	{
212     						out_info.path.block(iter,0,1,x.size()) = x.transpose();
213     						out_info.path(iter,x.size()) = F_x;
214     					}
215     				}
216     			} // end while
217     
218     			// Output info:
219     			out_info.final_sqr_err = F_x;
220     			out_info.iterations_executed = iter;
221     			out_info.last_err_vector = f_x;
222     			if (returnPath) out_info.path.setSize(iter,N+1);
223     
224     			MRPT_END
225     		}
226     
227     	}; // End of class def.
228     
229     
230     	typedef CLevenbergMarquardtTempl<mrpt::math::CVectorDouble> CLevenbergMarquardt;  //!< The default name for the LM class is an instantiation for "double"
231     
232     	} // End of namespace
233     } // End of namespace
234     #endif
