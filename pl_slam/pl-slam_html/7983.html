27     /** \brief Enumeration of available verbosity levels */
28     enum VerbosityLevel { LVL_DEBUG=0, LVL_INFO, LVL_WARN, LVL_ERROR };
29     
30     
31     /** \brief Versatile class for consistent logging and 
32      *        management of output messages
33      *
34      * COutputLogger is a versatile class for logging messages either to the
35      * terminal window or to an external file. Class instances can take messages in
36      * std::string using the log class methods 
37      *
38      * - Logger instance keeps the messages in an internal std::vector so that upon
39      *   request it can dump them either to the console or to an external file
40      *   altogether.  
41      * - The message, when printed in the terminal window, is **colored** according to
42      *   the logger's current verbosity/logging level (Logging level with which
43      *   the underlying TMsg instance was instantiatedd).  The available verbosity
44      *   levels as well as their corresponding colors are listed below:
45      *
46      *   + LVL_DEBUG => CONCOL_BLUE
47      *   + LVL_INFO  => CONCOL_NORMAL 
48      *   + LVL_WARN  => CONCOL_GREEN
49      *   + LVL_ERROR => CONCOL_RED
50      *
51      * - Logged messages are displayed in the screen if the current logger level is
52      *   higher than m_min_verbosity_level (logger ignores those messages
53      *   altogether). This can be used for filtering the output messages according
54      *   to their importance (e.g. show only error messages by issuing
55      *   setMinLoggingLevel(LVL_ERROR)).
56      *   \sa setLoggingLevel, setMinLoggingLevel
57      *
58      * \note By default every logged message is going to be dumped to the stadard
59      * output as well (if VerbisityLevel > m_min_verbosity_level). Unset \b
60      * print_message_automatically class variable if that's not the desired
61      * behavior
62      *
63      * \sa TMsg
64      * \ingroup mrpt_base_grp
65      */
66     class BASE_IMPEXP COutputLogger {
67     	public:
68     		/**
69     		 * \brief Construct a COutputLogger instance with the given name as the
70     		 * instance name.
71     		 *
72     		 * Call to this constructor can be used instead of first initializing the
73     		 * object and then explicitly setting the name like in the following case:
74     		 * \code
75     		 * COutputLogger a_logger;
76     		 * a_logger.setName("logger_name");
77     		 * \endcode
78     		 */
79     		COutputLogger(std::string name);
80     		/** \brief Default class constructor.
81     		 *
82     		 * Name of the logger is initialized to *Logger*
83     		 */
84     		COutputLogger();
85     		/**
86     		 * \brief Default class destructor
87     		 */
88     		virtual ~COutputLogger();  //!< virtual dtor (so we can derive classes from this one)
89     
90     		/** \brief Main method to add the specified message string to the logger.
91     		 *
92     		 * Message is printed in the terminal window if
93     		 * <b>print_message_automatically</b> is set to true. By default it is
94     		 * saved in the COutputLogger history.
95     		 *
96     		 * \sa logCond, logFmt
97     		 */
98     		void log(const std::string& msg_str);
99     		/** \brief Alternative of the previous method, which lets the user specify
100     		 *         a *temporary logging level*.
101     		 *
102     		 * Level will be reverted to the prior logging level
103     		 * after opration
104     		 * \sa logCond, logFmt
105     		 */
106     		void log(const std::string& msg_str, const VerbosityLevel& level);
107     		/** \brief Alternative logging method, *handy for usage in const
108     		 *         functions/methods*
109     		 *
110     		 * Used *solely for printing* the given string in the console (but does not
111     		 * store it internally)
112     		 * \sa logCond, logFmt
113     		 */
114     		void log(const std::string& msg_str, const VerbosityLevel& level) const;
115     		/** \brief Alternative logging method, handy for usage in const functions/methdos.
116     		 *
117     		 * Used solely for printing the given string in the console (but does not
118     		 * store it internally
119     		 * \sa logCond, logFmt
120     		 */
121     		void log(const std::string& msg_str) const;
122     		/** \brief Alternative logging method, which mimics the printf behavior.
123     		 *
124     		 * Handy for not having to first use mrpt::format to pass a std::string
125     		 * message to log
126     		 *
127     		 * \code
128     		 * // instead of:
129     		 * log(mrpt::format("Today is the %d of %s, %d", 15, "July", 2016));
130     		 *
131     		 * // one can use:
132     		 * logFmt("Today is the %d of %s, %d", 15, "July", 2016);
133     		 * \endcode
134     		 *
135     		 * \sa log, logCond, mrpt::utils:CDebugOutputCapable
136     		 */
137     		void logFmt(const char* fmt, ...) MRPT_printf_format_check(2,3);
138     		/** \brief Alternative logging method, which mimics the printf behavior.
139     		 *
140     		 * Handy for not having to first use mrpt::format to pass a std::string
141     		 * message to log. Used *solely for printing* the given string in the console
142     		 * (but does not store it internally)
143     		 *
144     		 * \sa log, logCond, mrpt::utils:CDebugOutputCapable
145     		 */
146     		void logFmt(const char* fmt, ...) const MRPT_printf_format_check(2,3);
147     		/** \brief Log the given message only if the condition is satisfied.
148     		 *
149     		 * Level will be reverted to the prior logging level
150     		 * after opration
151     		 *
152     		 * \sa log, logFmt
153     		 */
154     		void logCond(const std::string& msg_str, bool cond);
155     		/** \brief Log the given message only if the condition is satisfied.
156     		 *
157     		 * Used *solely for printing* the given string in the console (but does not
158     		 * store it internally)
159     		 *
160     		 * \sa log, logFmt
161     		 */
162     		void logCond(const std::string& msg_str, bool cond) const;
163     		/** \brief Log the given message only if the condition is satisfied.
164     		 *
165     		 * \sa log, logFmt
166     		 */
167     		void logCond(const std::string& msg_str, const VerbosityLevel& level, bool cond);
168     		/** \brief Log the given message only if the condition is satisfied.
169     		 *
170     		 * Used *solely for printing* the given string in the console (but does not
171     		 * store it internally)
172     		 * \sa log, logFmt
173     		 */
174     		void logCond(const std::string& msg_str, const VerbosityLevel& level, bool cond) const;
175     		/** \brief Set the name of the COutputLogger instance 
176     		 *
177     		 * \sa getName
178     		 */
179     		void setName(const std::string& name);
180     		/** \brief Return the name of the COutputLogger instance 
181     		 *
182     		 * \sa setName
183     		 */
184     		std::string getName() const;
185     		/** \brief Set the logging level that is going to be assigned in each message
186     		 *         logged from this point on.
187     		 *
188     		 * \sa setMinLoggingLevel, getLoggingLevel
189     		 */
190     		void setLoggingLevel(const VerbosityLevel& level = LVL_INFO);
191     		/** \brief Set the *minimum* logging level for which the incoming logs are going to
192     		 *         be taken into account.
193     		 *
194     		 * String messages with specified VerbosityLevel smaller than the min, will
195     		 * not be outputted to the screen and neither will a record of them be
196     		 * stored in by the COutputLogger instance
197     		 *
198     		 * \sa setLoggingLevel
199     		 */
200     		void setMinLoggingLevel(const VerbosityLevel& level = LVL_INFO);
201     		/** \brief Return the currently used VerbosityLevel 
202     		 *
203     		 * \sa setLoggingLevel
204     		 */
205     		VerbosityLevel getLoggingLevel() const;
206     		/** \brief Fill the provided string with the contents of the logger's
207     		 * history in std::string representation
208     		 *
209     		 */
210     		void getAsString(std::string* fname) const;
211     		/** \brief Get the history of COutputLogger instance in a string representation.
212     		 */
213     		std::string getAsString() const;
214     		/** \brief Write the contents of the COutputLogger instance to an external file.
215     		 *
216     		 * Upon call to this method, COutputLogger dumps the contents of all the
217     		 * logged commands so far to the specified external file.  By default the
218     		 * filename is set to ${LOGGERNAME}.log except if the fname parameter is
219     		 * provided
220     		 *
221     		 * \sa dumpToConsole, getAsString
222     		 */
223     		void writeToFile(const std::string* fname_in=NULL) const;
224     		/** \brief Dump the current contents of the COutputLogger instance in the
225     		 * terminal window.
226     		 *
227     			 * \sa writeToFile
228     		 */
229     		void dumpToConsole() const;
230     		/** \brief Return the last Tmsg instance registered in the logger history 
231     		 *
232     		 */
233     		std::string getLastMsg() const;
234     		/** \brief Fill inputtted string with the contents of the last message in history
235     		 */
236     		void getLastMsg(std::string* msg_str) const;
237     		/** \brief Reset the contents of the logger instance.
238     		 *
239     		 * Used *by default* when an instance of the class is initialized
240     		 */
241     		void reset();
242     
243     		/**
244     		 * \brief Struct responsible of holding information relevant to the message
245     		 *        (in std::string form) issued by the user.
246     		 *
247     		 * Upon TMsg initialization, instance fetches the name of the caller
248     		 * COutputLogger, as well as the VerbosityLevel and the
249     		 * mrpt::system::TTimeStamp of the message provided.
250     		 * The format of the message when this is printed / or written to an
251     		 * external file complies is given below:
252     		 *
253     		 * <center><em> [name | level | timestamp:] body </em></center>
254     		 */
255     		struct BASE_IMPEXP TMsg {
256     			/** \brief Class constructor that passes a message in std::string
257     			 * form as well as a reference to the COutputLogger that provided the
258     			 * current message
259     			 */
260     			TMsg(const std::string& msg, const COutputLogger& logger);
261     			/** \brief  Default Destructor */
262     			~TMsg();
263     
264     			/** \brief Return a string representation of the underlying message */
265     			std::string getAsString() const;
266     			/** \brief Fill the string with the contents of the underlying message in
267     			 * string representation */
268     			void getAsString(std::string* contents) const;
269     			/** \brief Write the message contents to the specified stream
270     			 *
271     			 * \sa getAsString
272     			 */
273     			void writeToStream(mrpt::utils::CStream& out) const;
274     			/** \brief Dump the message contents to the standard output
275     			 *
276     			 * \sa writeToStream
277     			 */
278     			void dumpToConsole() const;
279     			/** \brief Reset the contents of the TMsg instance */
280     			void reset();
281     			/** \brief Return the color corresponding to the given VerbosityLevel
282     			 *
283     			 * \sa getLoggingLevelName
284     			 */
285     			mrpt::system::TConsoleColor getConsoleColor(VerbosityLevel level) const;
286     			/** \brief Return the name corresponding to the given VerbosityLevel 
287     			 *
288     			 * \sa getConsoleColor
289     			 */
290     			std::string getLoggingLevelName(VerbosityLevel level) const; 
291     
292     			//
293     			// parameters of the message under construction
294     			//
295     			mrpt::system::TTimeStamp timestamp; /**< Timestamp of the message. */
296     			VerbosityLevel level; /**< Verbosity level of the message. */
297     			std::string name; /**< Name of the COutputLogger instance that called registered the message. */
298     			std::string body; /**< Actual content of the message. */
299     
300     			/** \brief Map from VerbosityLevels to their corresponding mrpt::system::TConsoleColor
301     			 *
302     			 * Handy for coloring the input based on the verbosity of the message
303     			 */
304      			std::map<VerbosityLevel, mrpt::system::TConsoleColor> m_levels_to_colors;
305     			/** \brief Map from VerbosityLevels to their corresponding names
306     			 *
307     			 * Handy for printing the current message VerbosityLevel along with the
308     			 * actual content
309     			 */
310     			std::map<VerbosityLevel, std::string> m_levels_to_names;
311     		};
312     
313     		/** \brief Set it to false in case you don't want the logged message to be
314     		 *         dumped to the output automatically. 
315     		 *
316     		 * By default it is set to true.
317     		 */
318     		bool print_message_automatically;
319     	private:
320     		/** Warn (at least for the first usages) that the logged messages are not
321     		 * stored by the COutputLogger instance since the call to
322     		 * COutputLogger::log is made inside a const method/function
323     		 */
324     		void warnForLogConstMethod() const;
325     
326     		/** Helper method for generating a std::string instance from printf-like
327     		 * arguments
328     		 */
329     		std::string generateStringFromFormat(const char* fmt, va_list argp) const;
330     
331     		std::string m_name;
332     		std::vector<TMsg> m_history;
333     		VerbosityLevel m_curr_level;
334     		/** \brief Provided messages with VerbosityLevel smaller than this value shall be ignored */
335     		VerbosityLevel m_min_verbosity_level;
336     		
337     		// how many times have I warned the user about using the const log method
338     		// if it passes a predefined threshold of times stop polluting the console
339     		static size_t m_times_for_log_const;
340     		static size_t m_times_for_log_const_thresh;
341     };
342     
343     } }  // END OF NAMESPACES

345     #endif /* end of include guard: COUTPUTLOGGER_H */
